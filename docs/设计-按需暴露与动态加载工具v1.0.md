### 0.3 prompt意义不大，也没有用上？

#### 问题诊断（您发现了一个关键问题！）

**现象**：LLM调用`get_current_time`时传入`path`/`file_path`而不是`timezone`

**根本原因**：
1. **description太长**（time_register.py:59-81），LLM可能没仔细看
2. **description只是"提示"，不是"强制约束"** - LLM被训练数据误导
3. **真正的强制约束在`parameters.properties`**（来自schema），但LLM可能没遵守

#### 现有机制的问题

从`react_schema.py:31-103`看，暴露给LLM的是**从schema生成**的：
```python
openai_tool = {
    "function": {
        "name": "get_current_time",
        "description": processed_description,  # ⚠️ LLM能看到，但可能不仔细看
        "parameters": {
            "properties": cleaned_properties,  # ✅ 强制约束，但LLM可能不遵守
            "required": required
        }
    }
}
```

**问题**：
- ❌ **LLM知道正确参数是`timezone`，但实际调用传入了`path`**
- ❌ **说明现有约束不够强**，LLM还是被训练数据误导
- ❌ **prompt（TOOL_DESCRIPTIONS）中的详细信息（使用场景、参数说明）没有被LLM有效利用**

#### 改进方案

**方案1：精简description，突出强制约束**
```python
# time_register.py 改进后
TIME_TOOL_DESCRIPTIONS = {
    "get_current_time": """获取当前系统时间。
    
    【强制参数约束】：
    ✅ 正确：{"timezone": "Asia/Shanghai", "format": "%Y-%m-%d %H:%M:%S"}
    ❌ 错误：{"path": "/Shanghai"}  ← 此参数不存在！
    ❌ 错误：{"file_path": "/Shanghai"}  ← 此参数不存在！
    
    【参数】：timezone(可选)、format(可选)、locale(可选）
    【示例】：{"timezone": "Asia/Shanghai"}
    """
}
```

**方案2：把详细的使用说明放到system prompt中**
```python
# base_react.py 的 system prompt
system_prompt = f"""
【可用工具详细说明】
- get_current_time: 获取当前时间
  使用场景：用户问"现在几点"、"今天星期几"
  正确调用：{"timezone": "Asia/Shanghai"}
  错误示例：{"path": "/Shanghai"} ← 无效参数！
...
"""
```

**3.3 公用Mixin代码设计**

```python
# -*- coding: utf-8 -*-
"""
ReactAgentMixin - ReAct Agent 公用逻辑混入类

提取FileReactAgent/TimeReactAgent的重复逻辑，
供新增的ShellReactAgent/NetworkReactAgent等复用。

Author: 小健 - 2026-05-06

from typing import Dict, Any, List, Optional, Callable
from app.services.agent.tool_executor import ToolExecutor
from app.services.agent.llm_strategies import TextStrategy, ToolsStrategy, ResponseFormatStrategy
from app.services.agent.llm_adapter import LLMAdapter

from app.services.tools.mixin import ToolLoaderMixin
from app.services.tools.registry import ToolCategory
from app.utils.logger import logger


class ReactAgentMixin(ToolLoaderMixin):
    """
    ReAct Agent 公用逻辑混入类
    
    使用方式：
        class ShellReactAgent(ReactAgentMixin, BaseAgent):
            def __init__(self, ...):
                super().__init__(...)
                self._init_tools_and_executor(tool_category)
                self._init_llm_strategies()

    def _init_tools_and_executor(self, tool_category: Optional[ToolCategory] = None):
        """初始化工具加载和执行器"""
        if tool_category:
            self._tools_dict = self.load_tools_by_category(tool_category)
        self.executor = ToolExecutor(self._tools_dict)
        logger.info(f"[{self.__class__.__name__}] 加载工具: {len(self._tools_dict)}个")
    
    def _init_llm_strategies(self):
        """初始化LLM调用策略"""
        self.text_strategy = TextStrategy()
        self.adapter = None
        self.use_function_calling = False
        self.openai_tools = []
        self.tools_strategy = None
        self.response_format_strategy = None
    
    def _init_session(self):
        """初始化session管理"""
        self.session_service = get_session_service()
        self._session_created_by_agent = False
    
    def _init_candidates(self, candidates: Optional[List[str]] = None):
        """初始化候选意图列表"""
        self._candidates = candidates if candidates else []
    
    # ===== 跨分类工具支持 =====
    
    def _get_tools_summary(self) -> str:
        """获取跨分类工具概要（每轮实时生成）"""
        from app.services.tools.registry import tool_registry
        return tool_registry.get_all_tools_summary(
            priority_category=self.tool_category or ToolCategory.FILE
        )
    
    def _build_candidates_hint(self) -> str:
        """构建候选意图提示"""
        if not self._candidates:
            return ""
        candidates_list = ", ".join(self._candidates)
        return (
            f"\n\n【候选意图】已识别出以下可能的意图类别: {candidates_list}。"
            "你可以根据实际任务需要，访问任意候选分类的工具。"
        )
    
    def _build_cross_tool_hint(self, category_name: str) -> str:
        """构建跨分类工具提示"""
        return (
            f"\n\n【注意】除了{category_name}工具，你还可以使用其他分类的工具。"
            "根据任务需要自由选择合适的工具，不受初始分类限制。"
        )
    
    def _build_system_prompt(self, category_name: str) -> str:
        """构建完整system prompt（base + candidates + cross）"""
        if hasattr(self, '_custom_system_prompt') and self._custom_system_prompt:
            return self._custom_system_prompt
        base = self.prompts.get_system_prompt()
        return base + self._build_candidates_hint() + self._build_cross_tool_hint(category_name)
    
    # ===== LLM调用 =====
    
    async def _call_llm_with_summary(self) -> str:
        """LLM调用统一入口（含工具概要注入+adapter策略选择）"""
        self.llm_call_count += 1
        logger.info(f"[LLM Counter] >>> LLM called, count: {self.llm_call_count}")
        
        last_message = self.conversation_history[-1]["content"]
        history_dicts = self.conversation_history[:-1]
        
        # 注入工具概要
        try:
            tools_summary = self._get_tools_summary()
            summary_msg = {"role": "system", "content": f"【当前可用工具列表】\n{tools_summary}"}
            history_dicts = list(history_dicts) + [summary_msg]
        except Exception as e:
            logger.warning(f"[ToolSummary] 注入工具概要失败: {e}")
        
        # adapter策略选择
        if self.adapter:
            strategy = await self.adapter.ensure_capability()
            if strategy.method == "response_format" and self.response_format_strategy:
                return await self.response_format_strategy.call(
                    llm_client=self.llm_client, message=last_message,
                    history_dicts=history_dicts, conversation_history=self.conversation_history)
            elif strategy.method == "tools" and self.tools_strategy:
                return await self.tools_strategy.call(
                    llm_client=self.llm_client, message=last_message,
                    history_dicts=history_dicts, conversation_history=self.conversation_history)
        
        # 默认文本模式
        return await self.text_strategy.call(
            llm_client=self.llm_client, message=last_message,
            history_dicts=history_dicts, conversation_history=self.conversation_history)
    
    # ===== Session管理 =====
    
    def _on_session_init(self, task: str, context=None):
        """Session初始化Hook"""
        if not self.task_id:
            self.task_id = self.session_service.create_session(
                agent_id=f"{self.__class__.__name__}-agent", task_description=task)
            self._session_created_by_agent = True
    
    def _on_after_loop(self):
        """循环结束后Hook - 关闭Session"""
        if self._session_created_by_agent and self.task_id and self.session_service:
            try:
                self.session_service.complete_session(self.task_id, success=True)
                self._session_created_by_agent = False
            except Exception as e:
                logger.error(f"Failed to complete session {self.task_id}: {e}")
```

#### 二、动态加载置信度阈值无降级策略

**原问题**：react_schema.py设计代码（1915-1917行）中，如果所有意图置信度都低于0.3，会导致初始无工具可用。

**修正方案**：在`get_tools_schema_for_intent_distribution()`中增加降级逻辑：

```python
# react_schema.py 新增函数（设计中）
def get_tools_schema_for_intent_distribution(
    intent_dist: Dict[str, float],
    top_n: int = 2,
    threshold: float = 0.3
) -> Tuple[List[Dict], str]:
    """
    按意图置信度分布返回混合工具Schema
    
    Returns:
        (tools_schema, brief_hint)
    """
    if not intent_dist:
        return [], ""
    
    sorted_intents = sorted(intent_dist.items(), key=lambda x: x[1], reverse=True)
    
    # ===== 降级策略（小健 2026-05-06） =====
    # 检查top_n个意图是否都低于阈值
    top_scores = [score for _, score in sorted_intents[:min(top_n, len(sorted_intents))]]
    
    if all(score < threshold for score in top_scores):
        # 降级策略1：只加载最高置信度的1个意图
        logger.warning(f"[Schema] 所有top-{top_n}意图置信度<={threshold}，降级为top-1")
        top_n = 1
        threshold = 0.1  # 放宽阈值
        
        if sorted_intents[0][1] < 0.05:
            # 降级策略2：连最高的都极低，给通用工具清单
            logger.warning(f"[Schema] 最高置信度<0.05，返回通用工具清单")
            return get_tools_schema_for_function_calling()[:10], "【通用工具清单】置信度过低，仅提供基础工具"
    
    # 正常加载
    schemas = []
    brief_intents = []
    
    for intent, score in sorted_intents[:top_n]:
        if score >= threshold:
            # 高置信度：加载完整Schema
            tools = get_tools_from_registry_by_category(intent)
            for tool in tools.values():
                schemas.append(_build_full_schema(tool))
        else:
            # 低置信度：加入简要清单
            brief_intents.append((intent, score))
    
    # 简要清单提示
    brief_hint = ""
    if brief_intents:
        brief_hint = f"【低置信度意图】{', '.join([f'{i}({s:.2f})' for i, s in brief_intents])} - 工具列表已省略"
    
    return schemas, brief_hint
```

```python
# agent/shell_react.py 修正
async def rollback(self, step_number=None) -> bool:
    """
    Shell命令无法回滚
    
    Returns:
        False - 表示回滚不可用，而非回滚失败
    """
```

#### 七、动态加载关键词检测可能误触发

**原问题**：设计代码中，用户说"不要执行shell命令"会因包含"shell"而误触发加载。

**修正方案**：

```python
# base_react.py 动态加载触发条件（修正版）

# 触发词映射（正向）
TRIGGER_MAP = {
    "shell": ["执行命令", "运行", "npm ", "pip ", "git ", "docker ", "python ", "node "],
    "network": ["下载", "请求", "访问", "http", "api", "网络"],
    "database": ["查询", "数据库", "sql", "table"],
    "desktop": ["截图", "窗口", "屏幕", "gui"],
    "file": ["读取文件", "写入文件", "目录", "文件夹"],
    "system": ["进程", "服务", "系统信息", "cpu", "内存"],
    "time": ["时间", "日期", "定时", "倒计时"],
}

# 否定词列表
NEGATIVE_WORDS = ["不要", "别", "不需要", "禁止", "不用", "无需", "禁止", "no ", "not ", "don't"]

def _should_trigger_dynamic_load(self, observation: str) -> Tuple[Optional[str], str]:
    """
    判断是否应该触发动态加载
    
    Returns:
        (intent, reason) 或 (None, "")
    """
    observation_lower = observation.lower()
    
    # 检查是否是否定语境
    is_negative_context = any(nw in observation_lower for nw in NEGATIVE_WORDS)
    
    for intent, keywords in TRIGGER_MAP.items():
        # 检查是否有触发词
        matched_keywords = [kw for kw in keywords if kw in observation_lower]
        
        if matched_keywords:
            # 检查是否是否定语境
            if is_negative_context:
                # 进一步检查：触发词前后是否有否定词
                for kw in matched_keywords:
                    kw_pos = observation_lower.find(kw)
                    # 检查触发词前20个字符是否有否定词
                    context_before = observation_lower[max(0, kw_pos-20):kw_pos]
                    if not any(nw in context_before for nw in NEGATIVE_WORDS):
                        # 不是否定语境，可以触发
                        return intent, f"Observation触发词: {kw}"
                # 所有匹配的触发词都被否定
                continue
            else:
                # 非否定语境，直接触发
                return intent, f"Observation触发词: {matched_keywords[0]}"
    
    return None, ""

# 使用示例
async def _on_observation(self, observation: str):
    intent, reason = self._should_trigger_dynamic_load(observation)
    if intent and intent not in self._loaded_categories:
        self.load_tools_by_intent(intent, reason=reason)
```

#### 二、task_id的真实含义澄清

**task_id在系统中的完整传递链**：

```
前端request_id
  → react_sse_wrapper._run_agent_sse_stream(task_id=...)
    → AgentFactory.create(task_id=...)
      → BaseAgent.__init__(self.task_id = task_id)    # 存储
      → FileTools(task_id=task_id)                    # 传递给file工具
        → FileSafetyService.write_operation(task_id)  # 按task_id记录操作
        → FileSafetyService.rollback_session(task_id) # 按task_id回滚
      → session_service.create_session(agent_id, task) # 按task_id创建统计记录
      → session_service.complete_session(task_id)      # 按task_id关闭统计记录
```

**task_id的本质**：**一次Agent执行任务的唯一标识**，不是"会话"，是"任务执行实例"。

| 概念 | 含义 | 生命周期 | 代码体现 |
|------|------|---------|---------|
| task_id | 一次任务执行实例 | Agent.run()开始→结束 | `BaseAgent.task_id`，safety/session用它做主键 |
| session | 多轮对话 | 跨多个task | 前端维护，当前代码中**不存在独立session概念** |

**结论**：代码中的`session_service`实际是`task_execution_tracker`（任务执行追踪器），命名有误导性。

##### 3.1 新增通用TaskTracker

```python
# -*- coding: utf-8 -*-
"""
TaskExecutionTracker - 任务执行追踪器

统一管理所有Agent的任务执行追踪。
- FileAgent: 追踪操作+统计+回滚（FileSafetyService + FileOperationSessionService）
- 其他Agent: 追踪执行统计（通用统计）

Author: 小健 - 2026-05-06
"""
from typing import Optional, Dict, Any
from app.utils.logger import logger


class TaskExecutionTracker:
    """
    任务执行追踪器（统一入口）
    
    根据 agent_id（意图类型）分发到对应的追踪服务：
    - file → FileOperationSessionService（操作追踪+统计+回滚）
    - 其他 → GenericTaskTracker（通用统计）
    """
    
    def __init__(self):
        self._trackers = {}  # {intent_type: tracker_instance}
        self._init_file_tracker()
        self._init_generic_tracker()
    
    def _init_file_tracker(self):
        """初始化file专用追踪器"""
        from app.services.agent.session import get_session_service
        self._trackers['file'] = get_session_service()
    
    def _init_generic_tracker(self):
        """初始化通用追踪器（非file意图使用）"""
        self._generic_tracker = GenericTaskTracker()
        for intent in ['time', 'shell', 'network', 'desktop', 'database', 
                       'system', 'document', 'code_execution']:
            self._trackers[intent] = self._generic_tracker
    
    def get_tracker(self, intent_type: str):
        """获取对应意图的追踪器"""
        return self._trackers.get(intent_type, self._generic_tracker)
    
    def create_task(self, task_id: str, agent_id: str, task_description: str):
        """创建任务追踪记录
        
        ⚠️ 修正-小沈 2026-05-06 14:37:45：
        FileOperationSessionService.create_session()不接受task_id参数，
        它自己生成sess-xxx。但safety用task_id记录操作。
        所以create_task只做统计记录，不应覆盖调用方的task_id。
        """
        tracker = self.get_tracker(agent_id)
        if tracker:
            try:
                tracker.create_session(agent_id=agent_id, task_description=task_description)
            except Exception as e:
                logger.warning(f"[TaskTracker] create_task失败: {e}")
    
    def complete_task(self, task_id: str, agent_id: str, success: bool = True):
        """完成任务追踪"""
        tracker = self.get_tracker(agent_id)
        if tracker:
            try:
                tracker.complete_session(task_id, success=success)
            except Exception as e:
                logger.warning(f"[TaskTracker] complete_task失败: {e}")


class GenericTaskTracker:
    """
    通用任务追踪器（非file意图使用）
    
    只做轻量统计，不记录操作详情（无法回滚）。
    存储到内存dict（可选：后续扩展到SQLite）。
    """
    
    def __init__(self):
        self._tasks: Dict[str, Dict[str, Any]] = {}  # {task_id: task_info}
    
    def create_session(self, agent_id: str, task_description: str) -> str:
        """创建任务记录"""
        from uuid import uuid4
        from datetime import datetime
        task_id = str(uuid4())
        self._tasks[task_id] = {
            "agent_id": agent_id,
            "task_description": task_description,
            "status": "running",
            "created_at": datetime.now().isoformat(),
        }
        logger.info(f"[GenericTaskTracker] 创建任务: {task_id}, agent={agent_id}")
        return task_id
    
    def complete_session(self, task_id: str, success: bool = True):
        """完成任务记录"""
        from datetime import datetime
        if task_id in self._tasks:
            self._tasks[task_id]["status"] = "completed" if success else "failed"
            self._tasks[task_id]["completed_at"] = datetime.now().isoformat()
            logger.info(f"[GenericTaskTracker] 完成任务: {task_id}, success={success}")


# 全局单例
_tracker_instance: Optional[TaskExecutionTracker] = None

def get_task_tracker() -> TaskExecutionTracker:
    """获取任务追踪器单例"""
    global _tracker_instance
    if _tracker_instance is None:
        _tracker_instance = TaskExecutionTracker()
```

### 3.3 按需暴露流程（修正-小沈 2026-05-05 19:53:50）

**关键参数说明**（补充-小沈 2026-05-05 19:53:50）：
| 参数 | 值 | 说明 |
|------|-----|------|
| **top_n** | 2 | 前2个高置信度意图给详细Schema，其余给简要Schema |
| **置信度阈值** | 0.3 | 低于此值的意图不加载（避免无意义加载） |
| **简要清单阈值** | 0.1 | 高于此值的低置信度意图才列入简要清单 |
| **详细Schema** | True | 高置信度意图使用完整参数模型（Pydantic生成） |
| **简要Schema** | False | 低置信度意图只给name+一句话描述 |

```python
# react_schema.py 新增（参数已明确）
def get_tools_schema_for_intent_distribution(
    intent_dist: Dict[str, float],
    top_n: int = 2,                    # ✅ 前2个高置信度给详细Schema
    full_schema_for_top: bool = True
) -> List[Dict]:
    """
    根据意图置信度分布，返回混合Schema
    
    Args:
        intent_dist: 意图置信度分布 {"file": 0.85, "time": 0.60, ...}
        top_n: 前N个高置信度意图给详细Schema（默认2）
        full_schema_for_top: True=详细Schema，False=简要Schema
    
    Returns:
        混合的工具Schema列表
    """
    # 1. 排序意图
    sorted_intents = sorted(intent_dist.items(), key=lambda x: x[1], reverse=True)
    
    openai_tools = []
    loaded_categories = set()
    
    # 2. 高置信度意图：详细Schema（阈值0.3）
    for intent, score in sorted_intents[:top_n]:
        if score < 0.3:  # ✅ 置信度阈值，低于0.3的不加载
            continue
        category = ToolCategory(intent)
        tools = get_tools_from_registry_by_category(category)
        for tool in tools:
            if full_schema_for_top:
                schema = _build_detailed_schema(tool)  # ✅ 详细Schema（Pydantic模型）
            else:
                schema = _build_brief_schema(tool, score)  # 简要Schema
            openai_tools.append(schema)
        loaded_categories.add(intent)
    
    # 3. 低置信度意图：只给简要清单（阈值0.1）
    brief_intents = []
    for intent, score in sorted_intents[top_n:]:
        if score >= 0.1:  # ✅ 只保留置信度>=0.1的
            brief_intents.append(f"- {intent}类工具（置信度{score:.2f}）")
    
    # 4. 在system prompt中添加
    if brief_intents:
        brief_tools_hint = "\n【可用但未加载详细说明的工具】\n" + "\n".join(brief_intents)
        brief_tools_hint += "\n如需使用，请说明需求，我将动态加载。"
    
    return openai_tools, brief_tools_hint
```

### 3.4 动态暴露流程（修正-小沈 2026-05-05 19:53:50）

**可靠性改进**（补充-小沈 2026-05-05 19:53:50）：
- ❌ 单纯关键词检测不可靠（可能误触发、漏触发）

```python
# base_react.py 新增（改进版）
class BaseReactAgent:
    def __init__(self, ...):
        self._tools_dict = {}           # 当前加载的工具函数
        self._tools_schema = []          # 当前给LLM的Schema
        self._loaded_categories = set()  # 已加载的分类
        self._intent_classifier = None   # 意图分类器（用于更可靠检测）
    
    def load_tools_by_intent(self, intent_type: str, reason: str = ""):
        """
        动态加载某个意图的工具
        
        Args:
            intent_type: 意图类型（file/time/shell/network等）
            reason: 加载原因（用于日志）
        """
        if intent_type in self._loaded_categories:
            return  # 已加载
        
        logger.info(f"[动态加载] 原因: {reason}，加载意图: {intent_type}")
        
        # 1. 获取该意图的工具
        category = ToolCategory(intent_type)
        new_tools = get_tools_from_registry_by_category(category)
        
        # 2. 添加到_tools_dict
        self._tools_dict.update(new_tools)
        self._loaded_categories.add(intent_type)
        
        # 3. 重新生成tools_schema（给LLM看）
        self._refresh_tools_schema()
        
        logger.info(f"[动态加载] 完成，新增{len(new_tools)}个工具，总计{len(self._tools_dict)}个")
    
    def _refresh_tools_schema(self):
        """重新生成tools_schema"""
        # 根据当前已加载的分类，生成Schema
        categories = list(self._loaded_categories)
        self._tools_schema = get_tools_schema_for_categories(categories)
        
        # 同时更新system prompt，告知LLM新增了工具
        self._update_system_prompt_with_tools()
    
    def _check_and_load_missing_tools(self, observation: str, llm_client=None):
        """
        在Observation阶段，检测是否需要新工具（改进版）
        
        改进方案：结合LLM判断 + 关键词检测
        """
        # 方案1：LLM判断（更可靠，推荐）
        if llm_client and self._intent_classifier:
            try:
                # 使用意图分类器重新检测
                result = self._intent_classifier.classify(observation)
                new_intent = result.get("intent")
                confidence = result.get("confidence", 0)
                
                if (new_intent and new_intent not in self._loaded_categories 
                    and confidence >= 0.3):
                    self.load_tools_by_intent(
                        new_intent, 
                        reason=f"LLM检测新意图: {new_intent}(置信度{confidence:.2f})"
                    )
                    return
            except Exception as e:
                logger.warning(f"[动态加载] LLM判断失败，降级到关键词: {e}")
        
        # 方案2：关键词检测（降级方案）
        trigger_keywords = {
            "network": ["ping ", "http", "下载", "网络", "url", "curl", "wget"],
            "desktop": ["截图", "截屏", "窗口", "鼠标点击", "键盘输入"],
            "shell": ["执行命令", "npm ", "pip ", "git ", "docker "],
            "database": ["数据库", "SQL", "查询", "SELECT", "INSERT"],
            "env": ["环境变量", "PATH", "JAVA_HOME"],
            "system": ["系统信息", "CPU", "内存", "磁盘"],
        }
        
        for intent, keywords in trigger_keywords.items():
            if any(kw in observation for kw in keywords):
                if intent not in self._loaded_categories:
                    self.load_tools_by_intent(
                        intent, 
                        reason=f"Observation关键词检测: {keywords}"
                    )
                    return  # 一次只加载一个，避免混乱
```

```python
        # ... 原来的Schema生成逻辑 ...
    return openai_tools


def _get_mixed_schema(
    intent_dist: Dict[str, float],
    top_n: int = 2
) -> List[Dict]:
    """混合Schema：高置信度详细，低置信度简要"""
    sorted_intents = sorted(intent_dist.items(), key=lambda x: x[1], reverse=True)
    
    openai_tools = []
    brief_intents = []
    
    # 高置信度：详细Schema
    for intent, score in sorted_intents[:top_n]:
        if score < 0.3:
            continue
        tools = get_tools_from_registry_by_category_string(intent)
        for tool in tools:
            openai_tools.append(_build_detailed_schema(tool))
    
    # 低置信度：收集简要清单
    for intent, score in sorted_intents[top_n:]:
        if score >= 0.1:
            brief_intents.append((intent, score))
    
    # 将简要清单作为特殊工具返回（或者不放在Function Calling中）
    if brief_intents:
        brief_desc = "【未加载详细说明的工具】" + "".join([f"\n- {i}({s:.2f})" for i, s in brief_intents])
        # 可以选择：1) 作为finish工具的description一部分；2) 放在system prompt中
    
    return openai_tools
```

```python
# ============== base_react.py 修改 ==============

class BaseReactAgent:
    def __init__(self, ..., intent_type: str = "file", intent_dist: Dict = None):
        # ... 原有逻辑 ...
        
        # 新增：保存意图信息
        self._intent_type = intent_type
        self._intent_dist = intent_dist or {}
        
        # 新增：初始只加载top-2意图的工具
        if intent_dist:
            self._loaded_categories = set()
            self._tools_dict = {}
            self._load_top_intents(intent_dist, top_n=2)
        else:
            # 兼容原有逻辑
            self._tools_dict = self._load_tools(tool_category)
    
    def _load_top_intents(self, intent_dist: Dict[str, float], top_n: int = 2):
        """根据置信度，加载top-N意图的工具"""
        sorted_intents = sorted(intent_dist.items(), key=lambda x: x[1], reverse=True)
        
        for intent, score in sorted_intents[:top_n]:
            if score >= 0.3:  # 阈值
                self.load_tools_by_intent(intent, reason=f"初始加载top-{top_n}意图")
    
    def load_tools_by_intent(self, intent_type: str, reason: str = ""):
        """动态加载某个意图的工具"""
        if intent_type in self._loaded_categories:
            return
        
        logger.info(f"[动态加载] 原因: {reason}，意图: {intent_type}")
        
        category = ToolCategory(intent_type)
        new_tools = get_tools_from_registry_by_category(category)
        
        self._tools_dict.update(new_tools)
        self._loaded_categories.add(intent_type)
        self._refresh_tools_schema()
    
    def _refresh_tools_schema(self):
        """重新生成tools_schema"""
        self._tools_schema = get_tools_schema_for_intent_distribution(
            {cat: 1.0 for cat in self._loaded_categories}  # 已加载的都给详细Schema
        )
    
    def _check_and_load_missing_tools(self, observation: str):
        """检测是否需要新工具"""
        trigger_map = {
            "network": ["ping", "http", "下载", "网络", "url"],
            "desktop": ["截图", "截屏", "窗口"],
            "shell": ["命令", "执行", "npm", "pip"],
            "database": ["数据库", "SQL", "查询"],
            "env": ["环境变量", "PATH"],
            "system": ["系统信息", "CPU", "内存"],
        }
        
        for intent, keywords in trigger_map.items():
            if any(kw in observation for kw in keywords):
                if intent not in self._loaded_categories:
                    self.load_tools_by_intent(intent, reason=f"Observation检测到: {observation[:50]}...")
                    return
```
