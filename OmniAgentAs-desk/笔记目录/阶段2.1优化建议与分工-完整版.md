```python
# 修改 backend/app/api/v1/chat.py

from fastapi import APIRouter, HTTPException
from fastapi.responses import StreamingResponse
import asyncio
import json

router = APIRouter()

@router.post("/chat/stream")
async def chat_stream(request: ChatRequest):
    """流式API - SSE (Server-Sent Events)"""
    
    async def generate():
        """生成SSE流"""
        
        # 1. 发送思考
        yield f"data: {json.dumps({\
            'type': 'thought',\
            'content': '正在分析任务...'\
        })}\n\n"
        
        await asyncio.sleep(0.5)
        
        # 2. 执行FileOperationAgent
        agent = FileOperationAgent(...)
        async for step in agent.run_stream(request.messages[-1].content):
            # 3. 发送每一步
            yield f"data: {json.dumps({\
                'type': step.type,\
                'step': step.step_number,\
                'thought': step.thought,\
                'action': step.action,\
                'action_input': step.action_input,\
                'observation': step.observation\
            })}\n\n"
            
            await asyncio.sleep(0.1)
        
        # 4. 发送结束
        yield f"data: {json.dumps({\
            'type': 'final',\
            'content': '任务完成'\
        })}\n\n"
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream"
    )
```

```typescript
export const chatApi = {
  // 现有同步API
  sendMessage: async (messages: ChatMessage[]) => {
    // ... 现有代码
  },
  
  // 新增：流式API
  sendMessageStream: async (
    messages: ChatMessage[],
    onStep: (step: any) => void,
    onComplete: (result: string) => void,
    onError: (error: string) => void
  ) => {
    try {
      const response = await fetch(`${API_BASE_URL}/chat/stream`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ messages, stream: true })
      });
      
      const reader = response.body!.getReader();
      const decoder = new TextDecoder();
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value);
        const lines = chunk.split('\n');
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            try {
              const data = JSON.parse(line.slice(6));
              
              if (data.type === 'thought' || 
                  data.type === 'action' || 
                  data.type === 'observation') {
                onStep(data);
              } else if (data.type === 'final') {
                onComplete(data.content);
              }
            } catch (e) {
              // 忽略解析错误
            }
          }
        }
      }
    } catch (error) {
      onError((error as Error).message);
    }
  },
  
  // ... 其他API
};
```

### 8.4 SSE流式API 设计与实现

#### 8.4.1 设计目标

- 实时展示ReAct执行步骤（思考→行动→观察）
- 支持任务中断功能
- 提供清晰的事件类型供前端处理

#### 8.4.2 事件类型定义

| 事件类型 | 说明 | 典型内容 |
|---------|------|---------|
| `thought` | AI思考过程 | "正在分析任务..." |
| `action` | 执行的操作步骤 | "检测到文件操作意图，开始执行..."、"执行文件操作..." |
| `observation` | 操作观察结果 | "安全检测通过" |
| `error` | 错误信息 | "危险操作需确认: xxx" |
| `interrupted` | 任务被中断 | "任务已被中断" |
| `final` | 最终完成 | 完整的执行结果 |

#### 8.4.3 实现逻辑流程

```
1. 接收请求
   ↓
2. 生成唯一task_id（UUID）
   ↓
3. 注册任务到running_tasks字典
   ↓
4. 发送"thought"事件 - 正在分析
   ↓
5. 检测是否为文件操作意图
   ↓
   ├─ 是文件操作 → 发送"action"事件 → 安全检测 → 发送"observation"
   │                            ↓
   │                     创建Agent执行 → 循环发送步骤
   │                            ↓
   │                     发送"final"完成
   │
   └─ 非文件操作 → 调用AI对话 → 发送"final"
   
6. 每个步骤检查中断状态
   ↓
7. 清理任务记录
```

| 第263-430行 | `chat_stream` SSE端点实现 |

#### 8.5.4 前端对接完整示例

```typescript
// SSE流式API对接示例

// 1. 发起请求
const response = await fetch('/api/v1/chat/stream', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ messages: [{ role: 'user', content: '帮我创建文件' }] })
});

// 2. 获取reader
const reader = response.body!.getReader();
const decoder = new TextDecoder();

// 3. 读取流
while (true) {
  const { done, value } = await reader.read();
  if (done) break;
  
  const chunk = decoder.decode(value);
  const lines = chunk.split('\n');
  
  for (const line of lines) {
    if (line.startsWith('data: ')) {
      try {
        const data = JSON.parse(line.slice(6));
        
        // 4. 根据type处理
        switch (data.type) {
          case 'thought':
            console.log('思考:', data.content);
            break;
          case 'action':
            console.log('操作:', data.content);
            break;
          case 'observation':
            console.log('观察:', data.content);
            break;
          case 'final':
            console.log('完成:', data.content);
            break;
          case 'interrupted':
            console.log('中断:', data.content);
            break;
          case 'error':
            console.error('错误:', data.content);
            break;
        }
      } catch (e) {
        // 忽略解析错误
      }
    }
  }
}
```
