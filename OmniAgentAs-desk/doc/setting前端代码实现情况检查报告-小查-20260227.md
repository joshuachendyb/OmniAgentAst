const handleFixConfig = async () => {
  setFixingConfig(true);
  setFixProgress({ current: 0, total: 100 });
  
  const progressInterval = setInterval(() => {
    setFixProgress(prev => {
      if (prev.current >= 100) {
        clearInterval(progressInterval);
        return prev;
      }
      return { ...prev, current: prev.current + 10 };
    });
