## 二、前端适配情况检查

### 2.1 错误处理适配

**检查位置**：`frontend/src/utils/sse.ts:477-487`

**当前代码**：
```typescript
case "error": {
  // 错误
  setIsReceiving(false);
  setIsConnected(false);
  disconnect();
  const errorMsg = step.content || "未知错误";  // ❌ 只处理了 content
  message.error(`流式传输错误：${errorMsg}`);
  onError?.(errorMsg);
  console.error("[SSE] 错误:", errorMsg);
  break;
}
```

**问题**：
1. ❌ **没有处理 error_type** - 无法根据错误类型显示不同提示
2. ❌ **没有处理 model/provider** - 丢失错误上下文信息
3. ❌ **没有传递完整错误对象** - 前端无法区分错误类型

**应该改为**：
```typescript
case "error": {
  // 错误
  setIsReceiving(false);
  setIsConnected(false);
  disconnect();
  
  // ✅ 处理新的错误格式
  const errorType = rawData.error_type || "unknown_error";
  const errorMsg = rawData.content || "未知错误";
  const model = rawData.model;
  const provider = rawData.provider;
  
  // ✅ 根据错误类型显示不同提示
  const errorMessages: Record<string, string> = {
    timeout_error: "请求超时，请检查网络连接",
    connection_error: "网络连接失败，请检查网络",
    http_error: "服务器响应异常，请稍后重试",
    value_error: "参数值错误，请检查输入",
    file_operation_error: "文件操作执行失败",
    unknown_error: errorMsg
  };
  
  const userMessage = errorMessages[errorType] || errorMsg;
  message.error(`流式传输错误：${userMessage}`);
  
  // ✅ 传递完整错误对象
  onError?.({
    type: errorType,
    message: userMessage,
    rawMessage: errorMsg,
    model,
    provider
  });
  
  console.error("[SSE] 错误:", {
    type: errorType,
    message: userMessage,
    model,
    provider
  });
  break;
}
```

**适配状态**：❌ **未适配**

## 三、NewChatContainer.tsx 适配检查

### 3.1 onError 回调适配

**检查位置**：`NewChatContainer.tsx:245-269`

**当前代码**：
```typescript
// onError - 流式错误
useCallback((error: string) => {
  console.error("SSE 流式错误:", error);
  message.error({
    content: `AI 响应失败：${error}`,
    duration: 5,
  });
  setMessages((prev) => {
    const lastMessage = prev[prev.length - 1];
    if (lastMessage && lastMessage.role === "assistant") {
      const updated = [...prev];
      updated[updated.length - 1] = {
        ...lastMessage,
        content: lastMessage.content || `**错误**: ${error}`,
        isError: true,
        isStreaming: false,
      };
      return updated;
    }
    return prev;
  });
  setLoading(false);
}, [])
```

**问题**：
1. ❌ **参数类型是 string** - 应该支持对象
2. ❌ **没有处理 error_type** - 无法根据类型显示不同 UI
3. ❌ **没有保存 model/provider** - 丢失错误上下文

**应该改为**：
```typescript
// onError - 流式错误
useCallback((error: string | {
  type: string;
  message: string;
  rawMessage: string;
  model?: string;
  provider?: string;
}) => {
  // ✅ 支持字符串和对象两种格式
  const errorObj = typeof error === 'string' 
    ? { type: 'unknown_error', message: error, rawMessage: error }
    : error;
  
  console.error("SSE 流式错误:", errorObj);
  
  message.error({
    content: `AI 响应失败：${errorObj.message}`,
    duration: 5,
  });
  
  setMessages((prev) => {
    const lastMessage = prev[prev.length - 1];
    if (lastMessage && lastMessage.role === "assistant") {
      const updated = [...prev];
      updated[updated.length - 1] = {
        ...lastMessage,
        content: lastMessage.content || `**错误**: ${errorObj.message}`,
        isError: true,
        isStreaming: false,
        errorType: errorObj.type,  // ✅ 保存错误类型
        model: errorObj.model,  // ✅ 保存模型
        provider: errorObj.provider,  // ✅ 保存提供商
      };
      return updated;
    }
    return prev;
  });
  setLoading(false);
}, [])
```

**适配状态**：❌ **未适配**
