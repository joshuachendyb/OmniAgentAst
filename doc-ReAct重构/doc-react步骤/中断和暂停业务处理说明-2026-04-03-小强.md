### 2.2 前端代码

```tsx
// NewChatContainer.tsx 第1624-1652行
const handleInterrupt = async () => {
  const taskIdToCancel = serverTaskId || taskId;  // ← 从 useSSE 获取
  console.log(`[中断] serverTaskId=${serverTaskId}, taskId=${taskId}, taskIdToCancel=${taskIdToCancel}`);
  
  if (taskIdToCancel) {
    try {
      message.info("正在中断任务...");
      
      // 先调用后端 API，再断开 SSE（修复：原来顺序反了）
      const result = await taskControlApi.cancel(taskIdToCancel, sessionId ?? undefined);
      console.log("[中断] cancel API 返回:", result);
      
      // 断开 SSE 连接
      disconnect(true);
      
      // 显示后端返回的具体消息
      if (result.message) {
        message.success(result.message);
      } else {
        message.success("任务中断请求已发送");
      }
    } catch (error) {
      message.error("发送中断请求失败: " + (error instanceof Error ? error.message : String(error)));
    }
  } else {
    message.warning("当前没有进行中的任务");
  }
};
```

### 2.3 后端代码

```python
# react_sse_wrapper.py 第579-619行
async def cancel_task(task_id: str, session_id: Optional[str] = None) -> Dict[str, Any]:
    # 1. 标记会话为中断（5分钟内禁止重连）
    if session_id:
        interrupted_sessions[session_id] = datetime.now()
        logger.info(f"[Session Interrupted] 会话 {session_id} 已标记为中断，5分钟内禁止重连")
    
    async with running_tasks_lock:
        logger.info(f"[TaskControl] 当前running_tasks数量: {len(running_tasks)}, keys: {list(running_tasks.keys())}")
        
        if task_id in running_tasks:
            task_info = running_tasks[task_id]
            task_info["cancelled"] = True
            task_info["status"] = "cancelled"
            
            # 强制关闭HTTP连接
            if "ai_service" in task_info and task_info["ai_service"]:
                ai_service = task_info["ai_service"]
                try:
                    ai_service.cancel()
                    logger.info(f"[Task Cancelled] 任务 {task_id} HTTP连接已强制关闭")
                except Exception as e:
                    logger.error(f"[Task Cancelled] 关闭HTTP连接失败: {e}")
            
            logger.info(f"[Task Cancelled] 任务 {task_id} 已标记为中断")
            return {"success": True, "message": f"任务 {task_id} 已中断"}
        else:
            logger.warning(f"[TaskControl] 任务 {task_id} 不在running_tasks中，可能已结束")
    
    if session_id:
        return {"success": True, "message": f"会话 {session_id} 已标记为中断（任务可能已完成）"}
    
    return {"success": False, "message": f"任务 {task_id} 不存在"}
```

### 2.4 API 定义

```typescript
// api.ts 第899-905行
cancel: async (taskId: string, sessionId?: string): Promise<TaskControlResponse> => {
  const url = sessionId 
    ? `/chat/stream/cancel/${taskId}?session_id=${sessionId}`
    : `/chat/stream/cancel/${taskId}`;
  const response = await api.post<TaskControlResponse>(url);
  return response.data;
},
```

## 五、taskId 获取流程

### 5.1 taskId 来源

```
后端 start 步骤返回
    ↓
sse.ts 第746行：setServerTaskId(rawData.task_id)
    ↓
useSSE hook 返回 serverTaskId
    ↓
NewChatContainer.tsx 第170行解构：serverTaskId
```

### 5.2 两个 taskId 的区别

| 变量 | 来源 | 含义 | 何时设置 |
|------|------|------|---------|
| `serverTaskId` | useSSE hook | 后端返回的任务 ID | start 步骤到达时 |
| `taskId` | useSSE hook | 前端生成的任务 ID | sendMessage 时 |

**优先级**：`serverTaskId || taskId`（优先使用后端返回的 ID）
