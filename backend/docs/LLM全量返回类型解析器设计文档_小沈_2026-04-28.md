## 三、解析器核心逻辑设计
**编写人**: 小沈
**时间**: 2026-04-28 13:40:00

### 3.1 解析流程图
```
输入output
  ├─ None/空值 → 返回None + 告警
  ├─ dict类型 → 直接调用`_create_action_result_from_dict`
  ├─ list类型 → 取最后一个元素，dict则解析，否则返回None
  └─ str类型
        ├─ 空字符串 → 返回None + 告警
        ├─ Markdown包裹 → 提取JSON后解析
        ├─ JSON数组字符串 → 解析后取最后一个元素
        ├─ 混合文本+JSON → 提取最后一个`{...}`解析
        └─ 标准JSON → 直接解析 / 非标准JSON修复后解析
              ↓
        `_create_action_result_from_dict` → 字段校验 → 类型转换 → 返回结果
```

### 3.2 核心函数设计
#### 3.2.1 `parse_react_response(output)`
- 功能：入口函数，路由所有LLM返回类型
- 每个分支标注：`处理的LLM返回类型` + `场景说明` + `署名+日期`

#### 3.2.2 `_parse_json_str(json_str)`
- 功能：解析JSON字符串，自动修复非标准格式
- 处理逻辑：标准解析 → 失败则修复（单引号转双引号、去尾逗号、去注释） → 再次解析

#### 3.2.3 `_fix_non_standard_json(json_str)`
- 功能：修复非标准JSON语法错误
- 修复项：单引号转双引号、去除对象/数组尾逗号、去除`//`单行注释

#### 3.2.4 `_create_action_result_from_dict(data)`
- 功能：从字典生成标准结果，统一字段校验
- 校验项：必填字段缺失检查、`content`类型强转、超长文本告警

## 六、代码实现（带全量注释）
**编写人**: 小沈
**时间**: 2026-04-28 13:55:00

完整代码见`D:\OmniAgentAs-desk\backend\app\services\agent\react_output_parser.py`，核心代码如下（带全量注释）：

```python
# react_output_parser.py
# 作者: 小沈
# 更新时间: 2026-04-28 12:45:00
# 说明: LLM全量返回类型解析器，覆盖所有已知LLM输出格式，无遗漏、无冗余
# 覆盖类型: None/空值、dict、list、标准JSON字符串、Markdown包裹JSON、混合文本+JSON、JSON数组、非标准JSON、字段缺失、content类型异常、超长文本(1000+/10000+)

import json
import re
import logging

logger = logging.getLogger(__name__)

def parse_react_response(output):
    """
    全量LLM返回类型解析器，覆盖所有已知LLM输出格式
    返回: ActionResult对象 / None（解析失败）
    """
    # 类型1: 处理输入为None/空值
    # 场景: LLM无返回、返回空值、网络超时等边界情况
    # 小沈-2026-04-28
    if not output:
        logger.warning("LLM返回为空，无法解析")
        return None

    # 类型2: 处理输入为dict对象
    # 场景: LLM返回被上游提前解析为字典（之前content丢失bug的根因）
    # 小沈-2026-04-28
    if isinstance(output, dict):
        logger.debug("检测到LLM返回为字典类型，直接处理")
        return _create_action_result_from_dict(output)

    # 类型3: 处理输入为list数组（对象维度）
    # 场景: LLM返回批量操作指令（如同时创建多个文件）
    # 小沈-2026-04-28
    if isinstance(output, list):
        if len(output) == 0:
            logger.warning("LLM返回空数组，无法解析")
            return None
        # 取最后一个有效操作指令（最新指令）
        last_item = output[-1]
        if isinstance(last_item, dict):
            logger.debug(f"检测到LLM返回为数组类型，取最后一个元素处理: {last_item.get('tool_name')}")
            return _create_action_result_from_dict(last_item)
        else:
            logger.warning(f"LLM返回数组元素类型异常: {type(last_item)}")
            return None

    # 类型4: 处理输入为字符串类型（最常见场景）
    # 小沈-2026-04-28
    if isinstance(output, str):
        # 去除前后空白字符
        stripped = output.strip()
        if not stripped:
            logger.warning("LLM返回为空字符串，无法解析")
            return None

        # 类型5: 处理Markdown包裹的JSON（```json\n{...}\n```）
        # 场景: LLM返回带markdown代码块的情况
        # 小沈-2026-04-28
        md_pattern = r'```json\s*\n?(.*?)\n?```'
        md_match = re.search(md_pattern, stripped, re.DOTALL)
        if md_match:
            json_str = md_match.group(1).strip()
            logger.debug("检测到Markdown包裹的JSON，提取后解析")
            return _parse_json_str(json_str)

        # 类型6: 处理JSON数组字符串（[{...}, {...}]）
        # 场景: LLM返回字符串格式的批量操作数组
        # 小沈-2026-04-28
        if stripped.startswith('[') and stripped.endswith(']'):
            try:
                data_list = json.loads(stripped)
                if isinstance(data_list, list) and len(data_list) > 0:
                    logger.debug(f"检测到字符串类型JSON数组，取最后一个元素处理")
                    return _create_action_result_from_dict(data_list[-1])
            except json.JSONDecodeError:
                pass

        # 类型7: 处理混合文本+JSON（提取最后一个完整{...}）
        # 场景: LLM返回解释性文字+JSON指令
        # 小沈-2026-04-28
        json_start = stripped.rfind('{')
        json_end = stripped.rfind('}') + 1
        if json_start != -1 and json_end != 0:
            # 验证是否是完整JSON（简单校验括号匹配）
            if stripped[json_start] == '{' and stripped[json_end-1] == '}':
                json_str = stripped[json_start:json_end]
                logger.debug("检测到混合文本+JSON，提取最后一个JSON解析")
                return _parse_json_str(json_str)

        # 类型4: 标准JSON字符串直接解析
        logger.debug("尝试标准JSON解析")
        return _parse_json_str(stripped)

    # 其他未知类型
    logger.error(f"LLM返回未知类型: {type(output)}，无法解析")
    return None


def _parse_json_str(json_str):
    """
    解析JSON字符串，自动修复常见语法错误
    参数: json_str - 待解析的JSON字符串
    返回: ActionResult对象 / None
    """
    # 先尝试标准解析
    try:
        data = json.loads(json_str)
        if isinstance(data, dict):
            return _create_action_result_from_dict(data)
        else:
            logger.warning(f"JSON解析结果非字典类型: {type(data)}")
            return None
    except json.JSONDecodeError as e:
        logger.warning(f"标准JSON解析失败: {e}，尝试修复非标准JSON")
        # 类型8: 修复常见JSON语法错误（单引号转双引号、去尾逗号、去注释）
        # 小沈-2026-04-28
        fixed_str = _fix_non_standard_json(json_str)
        try:
            data = json.loads(fixed_str)
            if isinstance(data, dict):
                logger.debug("非标准JSON修复后解析成功")
                return _create_action_result_from_dict(data)
        except json.JSONDecodeError as e2:
            logger.error(f"JSON修复后仍解析失败: {e2}，原始内容前200字符: {json_str[:200]}")
            return None


def _fix_non_standard_json(json_str):
    """
    修复非标准JSON语法错误：单引号转双引号、去除对象/数组尾逗号、去除注释
    小沈-2026-04-28
    """
    fixed = json_str
    # 1. 单引号转双引号（覆盖LLM常见错误）
    fixed = fixed.replace("'", '"')
    # 2. 去除对象/数组尾逗号（如{"a": 1,} → {"a": 1}）
    fixed = re.sub(r',\s*([}\]])', r'\1', fixed)
    # 3. 去除//单行注释
    fixed = re.sub(r'//.*', '', fixed)
    return fixed.strip()


def _create_action_result_from_dict(data):
    """
    从字典直接生成ActionResult，统一字段校验
    参数: data - LLM返回的字典对象
    返回: ActionResult对象 / None
    """
    # 类型9: 字段缺失校验
    # 小沈-2026-04-28
    required_fields = ["content", "tool_name", "tool_params"]
    for field in required_fields:
        if field not in data:
            logger.error(f"LLM返回字典缺失必填字段: {field}，原始数据: {data.get('tool_name')}")
            return None

    tool_name = data.get("tool_name")
    tool_params = data.get("tool_params")
    content = tool_params.get("content") if isinstance(tool_params, dict) else None
    reasoning = data.get("reasoning", "")

    # 类型10: content类型异常校验（强制转字符串）
    # 场景: LLM返回content为数字/布尔值/其他类型，避免下游工具报错
    # 小沈-2026-04-28
    if content is not None and not isinstance(content, str):
        logger.warning(f"tool_params.content类型异常: {type(content)}，强制转字符串")
        content = str(content)
        tool_params["content"] = content

    # 类型11: 超长文本校验（1000+字符、10000+字符）
    # 场景: 历史bug反馈的1000+字符content丢失，10000+字符性能风险
    # 小沈-2026-04-28
    if content:
        content_len = len(content)
        if content_len > 1000:
            logger.warning(f"检测到长文本content，长度: {content_len}")
        if content_len > 10000:
            logger.warning(f"检测到超长文本content，长度: {content_len}，超过10000字符阈值，注意解析性能")
        if content_len > 50000:
            logger.error(f"content长度超过50000字符，可能存在截断风险，长度: {content_len}")

    # 构造返回结果（适配项目ActionResult结构）
    return {
        "content": content,
        "tool_name": tool_name,
        "tool_params": tool_params,
        "reasoning": reasoning
    }
```
