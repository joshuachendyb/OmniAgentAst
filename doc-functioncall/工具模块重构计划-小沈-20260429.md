## 三、registry.py 核心设计【2026-04-29 更新】

### 3.1 主要功能

```python
# 全局单例
tool_registry = ToolRegistry()

# 【2026-04-29 标准注册方式】使用 input_model 参数
@register_tool(
    name="read_file",
    description="读取文件内容",
    category=ToolCategory.FILE,
    input_model=ReadFileInput,   # Pydantic 模型（自动生成 Schema）
    examples=[...]                # 使用示例（可选但推荐）
)
def tool_function(params):
    """实现"""
    pass

# 获取所有工具
get_registered_tools()

# 按分类获取
tool_registry.list_tools(category=ToolCategory.FILE)
```

> **【重要】** 必须使用 `input_model` 参数传入 Pydantic 模型类，禁止手动编写 `input_schema` 字典。

### 9.3 {type}_register.py 标准模板

```python
# 【小健 2026-04-29】强制规范：新增工具类型必须按此规范注册
# 必须从 {type}_schema 导入 Pydantic 模型，禁止手动编写 input_schema
from app.services.tools.{type}.{type}_schema import (
    Tool1Input,
    Tool2Input,
    # ... 其他模型
)
from app.services.tools.registry import register_tool, ToolCategory, tool_registry
from app.services.tools.{type}.{type}_tools import {Type}Tools

# 工具描述
{TYPE}_TOOL_DESCRIPTIONS = {
    "tool1": "工具1描述",
    "tool2": "工具2描述",
}

# 模型映射（必须）
{TYPE}_TOOL_INPUT_MODELS = {
    "tool1": Tool1Input,
    "tool2": Tool2Input,
}

# 使用示例（推荐）
{TYPE}_TOOL_EXAMPLES = {
    "tool1": [{"param": "value"}],
    "tool2": [...],
}

# 统一注册函数
def _register_{type}_tools():
    ft = {Type}Tools()
    tool_methods = {
        "tool1": ft.tool1,
        "tool2": ft.tool2,
    }
    
    for name, method in tool_methods.items():
        tool_registry.register(
            name=name,
            description={TYPE}_TOOL_DESCRIPTIONS.get(name, ""),
            category=ToolCategory.{TYPE},
            implementation=method,
            input_model={TYPE}_TOOL_INPUT_MODELS.get(name),  # 必须使用 input_model
            examples={TYPE}_TOOL_EXAMPLES.get(name, [])
        )

_register_{type}_tools()
```
