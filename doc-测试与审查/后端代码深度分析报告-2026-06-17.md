### 3.1 第一遍：目录分布与结构问题

```python
"""Step类型定义 — 合并所有Step数据类 — 小沈 2026-06-17"""
from dataclasses import dataclass, field
from typing import Any, Dict, Optional

@dataclass
class ReasoningStep:
    """Step基类"""
    step: int
    type: str = ""

@dataclass
class ChunkStep(ReasoningStep):
    content: str = ""
    is_reasoning: bool = False

@dataclass
class ErrorStep(ReasoningStep):
    error_type: str = ""
    error_message: str = ""
    recoverable: bool = False

@dataclass
class FinalStep(ReasoningStep):
    response: str = ""
    thought: str = ""

@dataclass
class MetaStep(ReasoningStep):
    type: str = ""
    message: str = ""
    data: Dict = field(default_factory=dict)

@dataclass
class ThoughtStep(ReasoningStep):
    content: str = ""
    tool_name: str = ""
    tool_params: Dict = field(default_factory=dict)
    thought: str = ""
    reasoning: str = ""

@dataclass
class ToolStep(ReasoningStep):
    tool_name: str = ""
    tool_params: Dict = field(default_factory=dict)
    execution_result: Any = None
    step_type: str = ""
    observation: str = ""
    execution_status: str = ""
    code: str = ""
    warning: str = None
    attachment: Any = None
    next_actions: Any = None
    return_direct: bool = False
```

```python
class Config:
    """配置管理类"""

    def get_tool_aliases(self, tool_name: str) -> Optional[Dict[str, str]]:
        """获取工具参数别名"""
        tools_config = self._config_data.get('tools', {})
        aliases = tools_config.get('aliases', {})
```

### 4.12 问题12: _trim_to_budget 保持现状

**分析结论**: 当前实现已清晰，保持原样

**理由**:
1. 当前50行代码逻辑连贯，拆分反而增加认知负担
2. 已有清晰的注释说明算法逻辑
3. 违反KISS：过度拆分不是简化，是复杂化
4. 函数内部逻辑紧密相关，拆分会破坏内聚性

**保持现状**:
```python
def _trim_to_budget(self, obs_list, assistant_msgs, budget):
    """FC-only: 从最新往最旧扫,按配对收集,简洁高效
    
    策略: 从最后一条消息往前遍历,遇到tool就找其配对assistant一起保留,
    遇到独立消息直接保留,直到budget用完。剩余的全部丢弃。
    小欧 2026-06-16: 保留每种工具的首次observation，防止LLM忘记已搜索过。
    """
    # 1. 构建 tool_call_id → assistant 映射
    tool_to_assistant = {}
    for msg in assistant_msgs:
        for tc in (msg.get("tool_calls") or []):
            if tc.get("id"):
                tool_to_assistant[tc["id"]] = msg

    # 2. 按原始顺序排列
    original_order = {id(msg): i for i, msg in enumerate(self.conversation_history)}
    all_msgs = sorted(obs_list + assistant_msgs, key=lambda m: original_order.get(id(m), 0))

    # 3. 识别每种工具的首次observation
    first_tool_obs = {}
    for msg in obs_list:
        tool_call_id = msg.get("tool_call_id", "")
        if tool_call_id:
            assistant = tool_to_assistant.get(tool_call_id)
            if assistant:
                for tc in assistant.get("tool_calls", []):
                    tool_name = tc.get("function", {}).get("name", "")
                    if tool_name and tool_name not in first_tool_obs:
                        first_tool_obs[tool_name] = msg

    # 4. 从后往前收集
    kept = []
    used_chars = 0
    i = len(all_msgs) - 1
    while i >= 0:
        msg = all_msgs[i]
        tc_id = msg.get("tool_call_id", "")

        if msg.get("role") == "tool" and tc_id and tc_id in tool_to_assistant:
            asst = tool_to_assistant[tc_id]
            pair_chars = self._total_chars([asst, msg])
            is_first_obs = msg in first_tool_obs.values()
            if is_first_obs or used_chars + pair_chars <= budget:
                kept.append(msg)
                kept.append(asst)
                used_chars += pair_chars
            i -= 1
            continue

        msg_chars = self._total_chars([msg])
        if used_chars + msg_chars <= budget:
            kept.append(msg)
            used_chars += msg_chars
        else:
            break
        i -= 1

    kept.reverse()
    return kept
```

**结论**: 此问题不修改，保持现有实现
