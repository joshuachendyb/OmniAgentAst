## 一、项目概述

### 1.1 项目背景

当前CRSS（Command Risk Scoring System）评分系统存在以下问题：
1. 评分公式设计缺陷，导致误判率高
2. 模式匹配过于简单，缺乏上下文理解
3. 关键词匹配宽泛，容易误匹配
4. 缺乏命令语义解析能力
5. 评分结果缺乏可解释性
6. 配置缺乏灵活性

### 1.2 项目目标

**总体目标：** 提升CRSS评分系统的准确性、可维护性和用户体验

**具体目标：**
1. 将误判率降低50%以上
2. 提供详细的评分解释
3. 支持动态配置更新
4. 平均响应时间保持在1ms以内
5. 提高系统可扩展性

### 1.3 项目范围

**包含：**
- 评分算法优化
- 模式匹配改进
- 关键词匹配增强
- 命令语义解析
- 配置管理系统
- 测试框架

**不包含：**
- 前端界面重构
- 数据库存储
- 用户管理系统

### 2.2 核心算法改进

#### 2.2.1 评分公式v2.0

```python
def calculate_risk_score_v2(command: str, parsed_info: dict = None) -> RiskAssessment:
    """
    改进的CRSS评分算法v2.0
    
    公式：最终分数 = 基础分数 × 平滑系数 + 风险加成
    
    基础分数 = 操作类型分数 × 类型权重 + 操作对象分数 × 对象权重 + 交互项
    平滑系数 = 分段平滑(影响范围系数)
    风险加成 = 高风险组合额外加分
    """
    # 1. 解析命令（如果未提供）
    if parsed_info is None:
        parsed_info = parse_command_v2(command)
    
    # 2. 获取各维度分数
    type_score = get_operation_type_score(parsed_info['operation_type'])
    target_score = get_operation_target_score(parsed_info['operation_target'])
    scope_multiplier = get_scope_multiplier(parsed_info['impact_scope'])
    
    # 3. 计算基础分数（加权平均）
    type_weight = 0.6  # 操作类型权重（最重要）
    target_weight = 0.3  # 操作对象权重
    interaction_weight = 0.1  # 交互项权重
    
    base_score = (
        type_score * type_weight +
        target_score * target_weight +
        calculate_interaction(type_score, target_score) * interaction_weight
    )
    
    # 4. 计算平滑系数（分段平滑）
    smoothed_multiplier = smooth_scope_multiplier(scope_multiplier)
    
    # 5. 计算风险加成
    risk_bonus = calculate_risk_bonus(
        parsed_info['operation_type'],
        parsed_info['operation_target'],
        parsed_info['impact_scope']
    )
    
    # 6. 计算最终分数
    raw_score = base_score * smoothed_multiplier + risk_bonus
    final_score = min(10, max(0, round(raw_score)))
    
    # 7. 构建详细结果
    return RiskAssessment(
        score=final_score,
        level=get_risk_level(final_score),
        message=get_risk_message(final_score),
        details={
            'operation_type': parsed_info['operation_type'],
            'operation_target': parsed_info['operation_target'],
            'impact_scope': parsed_info['impact_scope'],
            'type_score': type_score,
            'target_score': target_score,
            'scope_multiplier': scope_multiplier,
            'base_score': base_score,
            'smoothed_multiplier': smoothed_multiplier,
            'risk_bonus': risk_bonus,
            'confidence': calculate_confidence(parsed_info),
            'parsed_info': parsed_info
        }
    )
```

#### 2.2.2 分段平滑函数

```python
def smooth_scope_multiplier(multiplier: float) -> float:
    """
    分段平滑函数
    低风险范围：线性增长
    高风险范围：对数增长
    """
    if multiplier <= 1.5:
        # 低风险范围：线性增长，斜率0.8
        return 1 + (multiplier - 1) * 0.8
    elif multiplier <= 2.5:
        # 中风险范围：平方根增长
        return 1.4 + (multiplier - 1.5) ** 0.5
    else:
        # 高风险范围：对数增长，使用log3减缓增长
        return 1.8 + math.log(multiplier, 3)
```

def calculate_confidence(parsed_info: dict) -> float:
    """
    计算评分置信度
    考虑因素：匹配质量、上下文一致性、历史准确率
    """
    confidence = 1.0
    
    # 1. 匹配质量（0.7-1.0）
    match_quality = calculate_match_quality(parsed_info)
    confidence *= match_quality
    
    # 2. 上下文一致性（0.8-1.0）
    context_consistency = calculate_context_consistency(parsed_info)
    confidence *= context_consistency
    
    # 3. 历史准确率（0.9-1.0）
    historical_accuracy = get_historical_accuracy(parsed_info)
    confidence *= historical_accuracy
    
    return min(1.0, max(0.0, confidence))

def calculate_match_quality(parsed_info: dict) -> float:
    """计算匹配质量"""
    quality = 1.0
    
    # 操作类型匹配质量

        command_lower = command.lower()
        matches = []

        context = self.context_analyzer.analyze(command)
        
        # 检查所有模式
        for target_type, config in TARGET_WEIGHTS.items():
            for pattern in config['patterns']:
                match = re.search(pattern, command_lower, re.IGNORECASE)
                if match:
                    confidence = self._calculate_match_confidence(
                        pattern, match, command_lower, context, target_type
                    )
                    matches.append({
                        'type': target_type,
                        'pattern': pattern,
                        'match': match.group(),
                        'start': match.start(),
                        'end': match.end(),
                        'confidence': confidence
                    })
        
        # 如果没有匹配，返回默认值
        if not matches:
            return 'USER', 0.5, []
        
        # 选择置信度最高的匹配
        best_match = max(matches, key=lambda x: x['confidence'])
        
        # 检查是否有冲突的高置信度匹配
        high_confidence_matches = [m for m in matches if m['confidence'] > 0.7]
        if len(high_confidence_matches) > 1:
            # 多个高置信度匹配，需要进一步分析
            return self._resolve_conflicts(high_confidence_matches, context)
        
        return best_match['type'], best_match['confidence'], matches
    
    def _calculate_match_confidence(self, pattern: str, match, command: str, context: dict, target_type: str) -> float:
        """计算匹配置信度"""
        confidence = 1.0
        
        # 1. 匹配长度比例（0.5-1.0）
        match_length = len(match.group())
        command_length = len(command)
        length_ratio = match_length / max(command_length, 1)
        confidence *= max(0.5, min(1.0, length_ratio * 2))
        
        # 2. 位置权重（0.8-1.2）
        position_weight = self._calculate_position_weight(match.start(), match.end(), command_length)
        confidence *= position_weight
        
        # 3. 模式特异性（0.9-1.3）
        specificity = self._calculate_pattern_specificity(pattern)
        confidence *= specificity
        
        # 4. 上下文一致性（0.7-1.3）
        context_weight = self._calculate_context_weight(target_type, context)
        confidence *= context_weight
        
        # 5. 历史准确率（0.9-1.1）
        historical_weight = self._get_historical_weight(pattern, target_type)
        confidence *= historical_weight
        
        return min(1.0, max(0.0, confidence))

class ContextAnalyzer:
    """上下文分析器"""
    
    def analyze(self, command: str) -> dict:
        """分析命令上下文"""
        context = {
            'has_system_keywords': False,
            'has_project_keywords': False,
            'has_user_keywords': False,
            'has_temp_keywords': False,
            'operation_verbs': [],
            'paths': [],
            'quantifiers': [],
            'negations': [],
            'confidence_factors': {}
        }
        
        # 提取操作动词
        context['operation_verbs'] = self._extract_operation_verbs(command)
        
        # 提取路径
        context['paths'] = self._extract_paths(command)
        
        # 提取量词
        context['quantifiers'] = self._extract_quantifiers(command)
        
        # 提取否定词
        context['negations'] = self._extract_negations(command)
        
        # 分析关键词
        command_lower = command.lower()
        context['has_system_keywords'] = any(
            kw in command_lower for kw in ['windows', 'system32', '/bin', '/etc', '注册表', '密码']
        )
        context['has_project_keywords'] = any(
            kw in command_lower for kw in ['src/', 'backend/', 'frontend/', '.git', 'package.json']
        )
        
        return context
    
    def _extract_operation_verbs(self, command: str) -> list:
        """提取操作动词"""
        verbs = []
        command_lower = command.lower()
        
        operation_keywords = {
            'DELETE': ['删除', '移除', '删掉', '清除', 'rm', 'del'],
            'COPY': ['复制', '拷贝', 'copy', 'cp'],
            'MOVE': ['移动', '搬到', '移到', 'move', 'mv'],
            'READ': ['查看', '看', '显示', '列出', 'ls', 'dir', 'cat'],
            'EXEC': ['执行', '运行', '启动', 'exec', 'run', 'start'],
        }
        
        for op_type, keywords in operation_keywords.items():
            for keyword in keywords:
                # 使用词边界匹配
                pattern = r'(?<!\w)' + re.escape(keyword) + r'(?!\w)'
                if re.search(pattern, command_lower):
                    verbs.append({
                        'type': op_type,
                        'keyword': keyword,
                        'position': command_lower.find(keyword)
                    })
        
        return sorted(verbs, key=lambda x: x['position'])

### 2.4 有限状态机命令解析器

```python
class CommandParserFSM:
    """基于有限状态机的命令解析器"""
    
    def __init__(self):
        # 状态定义
        self.states = {
            'START': self._state_start,
            'OPERATION': self._state_operation,
            'SOURCE': self._state_source,
            'TARGET': self._state_target,
            'OPTIONS': self._state_options,
            'END': self._state_end
        }
        
        # 状态转移表
        self.transitions = {
            'START': ['OPERATION', 'END'],
            'OPERATION': ['SOURCE', 'TARGET', 'OPTIONS', 'END'],
            'SOURCE': ['TARGET', 'OPTIONS', 'END'],
            'TARGET': ['OPTIONS', 'END'],
            'OPTIONS': ['END'],
            'END': []
        }
        
        # 解析结果
        self.result = {
            'operation': None,
            'sources': [],
            'targets': [],
            'options': [],
            'direction': None,
            'quantity': 'single',
            'confidence': 1.0,
            'errors': []
        }
        
        # 当前状态
        self.current_state = 'START'
        self.tokens = []
    
    def parse(self, command: str) -> dict:
        """解析命令"""
        # 1. 分词
        self.tokens = self._tokenize(command)
        
        # 2. 状态机解析
        token_index = 0
        while self.current_state != 'END' and token_index < len(self.tokens):
            token = self.tokens[token_index]
            next_state = self.states[self.current_state](token)
            
            if next_state in self.transitions[self.current_state]:
                self.current_state = next_state
                token_index += 1
            else:
                # 无效状态转移，记录错误
                self.result['errors'].append(
                    f"无效状态转移: {self.current_state} -> {next_state} (token: {token})"
                )
                # 尝试恢复
                self.current_state = self._try_recover(token_index)
                if self.current_state == 'END':
                    break
        
        # 3. 后处理
        self._post_process()
        
        return self.result
    
    def _tokenize(self, command: str) -> list:
        """分词"""
        tokens = []
        
        # 中文分词（简单版本）
        import jieba
        words = jieba.lcut(command)
        
        # 合并路径和选项
        current_token = ''
        for word in words:
            if self._is_path_word(word) or self._is_option_word(word):
                current_token += word
            else:
                if current_token:
                    tokens.append(current_token)
                    current_token = ''
                tokens.append(word)
        
        if current_token:
            tokens.append(current_token)
        
        return tokens
    
    def _state_start(self, token: str) -> str:
        """起始状态"""
        if self._is_operation_token(token):
            self.result['operation'] = self._parse_operation(token)
            return 'OPERATION'
        elif self._is_source_token(token):
            self.result['sources'].append(token)
            return 'SOURCE'
        else:
            self.result['errors'].append(f"起始状态遇到无效token: {token}")
            return 'END'
    
    def _state_operation(self, token: str) -> str:
        """操作状态"""
        if self._is_source_token(token):
            self.result['sources'].append(token)
            return 'SOURCE'
        elif self._is_target_token(token):
            self.result['targets'].append(token)
            return 'TARGET'
        elif self._is_option_token(token):
            self.result['options'].append(token)
            return 'OPTIONS'
        else:
            return 'END'
    
    # ... 其他状态处理函数
    
    def _post_process(self):
        """后处理"""
        # 确定操作方向
        if self.result['operation'] in ['COPY', 'MOVE']:
            self.result['direction'] = 'write'
        elif self.result['operation'] == 'DELETE':
            self.result['direction'] = 'delete'
        elif self.result['operation'] == 'READ':
            self.result['direction'] = 'read'
        
        # 确定操作数量
        quantifiers = ['所有', '全部', '批量', '多个', '几个', '一些']
        for token in self.tokens:
            if token in quantifiers:
                self.result['quantity'] = 'batch'
                break
        
        # 计算置信度
        self.result['confidence'] = self._calculate_confidence()
    
    def _calculate_confidence(self) -> float:
        """计算解析置信度"""
        confidence = 1.0
        
        # 操作类型置信度
        if self.result['operation']:
            confidence *= 0.9
        else:
            confidence *= 0.5
        
        # 源路径置信度
        if self.result['sources']:
            confidence *= 0.8
        else:
            confidence *= 0.6
        
        # 错误数量影响
        error_penalty = max(0, 1 - len(self.result['errors']) * 0.1)
        confidence *= error_penalty
        
        return min(1.0, max(0.0, confidence))
```

            if k not in config:
                config[k] = {}
            config = config[k]
        
        # 设置值
        old_value = config.get(keys[-1])
        config[keys[-1]] = value
        
        # 验证配置
        if validate:
            try:
                self._validate_config()
            except ValueError as e:
                # 验证失败，恢复旧值
                if old_value is not None:
                    config[keys[-1]] = old_value
                else:
                    del config[keys[-1]]
                raise e
        
        # 保存配置
        self._save_config()
    
    def _save_config(self):
        """保存配置"""
        with open(self.config_file, 'w', encoding='utf-8') as f:
            yaml.dump(self.config, f, allow_unicode=True, default_flow_style=False)
    
    def _notify_listeners(self):
        """通知监听器配置已更新"""
        # 这里可以添加事件通知机制
        logger.info("配置已更新，通知相关组件")
