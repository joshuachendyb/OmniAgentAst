```python
# app/tools/base.py
from abc import ABC, abstractmethod
from typing import Any, Dict, Optional
from pydantic import BaseModel, Field

class ToolParameter(BaseModel):
    """工具参数定义"""
    type: str
    description: str
    required: bool = True
    default: Optional[Any] = None

class ToolDefinition(BaseModel):
    """工具定义Schema"""
    name: str
    description: str
    parameters: Dict[str, ToolParameter]
    returns: Dict[str, Any]
    danger_level: str = "low"  # low, medium, high
    examples: list = []

class ToolResult(BaseModel):
    """工具执行结果"""
    success: bool
    data: Optional[Any] = None
    error: Optional[str] = None
    execution_time: float = 0.0

class BaseTool(ABC):
    """工具基类"""
    
    def __init__(self):
        self.definition = self._get_definition()
    
    @abstractmethod
    def _get_definition(self) -> ToolDefinition:
        """返回工具定义"""
        pass
    
    @abstractmethod
    async def execute(self, **kwargs) -> ToolResult:
        """执行工具"""
        pass
    
    def validate_params(self, params: Dict[str, Any]) -> bool:
        """验证参数是否合法"""
        # 检查必需参数
        for name, param in self.definition.parameters.items():
            if param.required and name not in params:
                return False
        return True
```

#### 2.2.4 ReAct 循环核心

```python
# app/core/react_executor.py
from typing import List, Dict, Any, Optional
import json

class ReActExecutor:
    """ReAct 执行器：思考-行动-观察循环"""
    
    def __init__(self, ai_service, tool_registry):
        self.ai = ai_service
        self.tools = tool_registry
        self.max_iterations = 10  # 最大迭代次数，防止死循环
        self.retry_attempts = 3   # 错误重试次数
    
    async def execute(self, user_input: str, history: List[Dict] = None) -> Dict[str, Any]:
        """
        执行 ReAct 循环
        
        Args:
            user_input: 用户输入
            history: 历史消息
            
        Returns:
            执行结果
        """
        iterations = 0
        context = {
            "user_input": user_input,
            "thoughts": [],
            "actions": [],
            "observations": []
        }
        
        while iterations < self.max_iterations:
            iterations += 1
            
            # Step 1: Thought (思考)
            thought = await self._think(context, history)
            context["thoughts"].append(thought)
            
            # 检查是否应该结束
            if thought.get("should_stop", False):
                return {
                    "success": True,
                    "answer": thought.get("answer", ""),
                    "context": context
                }
            
            # Step 2: Action (行动)
            action = thought.get("action")
            if not action:
                continue
                
            context["actions"].append(action)
            
            # Step 3: Observation (观察)
            observation = await self._act(action)
            context["observations"].append(observation)
            
            # 如果执行失败，尝试重试
            if not observation.get("success", False):
                for attempt in range(self.retry_attempts):
                    retry_observation = await self._retry_action(action, observation, attempt)
                    if retry_observation.get("success", False):
                        observation = retry_observation
                        break
        
        # 达到最大迭代次数，强制结束
        return {
            "success": False,
            "error": "达到最大迭代次数",
            "context": context
        }
    
    async def _think(self, context: Dict, history: List[Dict]) -> Dict:
        """
        思考阶段：调用 AI 决定下一步行动
        
        返回格式：
        {
            "thought": "思考过程...",
            "action": {"tool": "tool_name", "params": {...}},
            "should_stop": false,
            "answer": ""  # 只有在 should_stop=true 时才有值
        }
        """
        # 构建 prompt
        prompt = self._build_think_prompt(context, history)
        
        # 调用 AI
        response = await self.ai.chat(prompt)
        
        # 解析响应
        return self._parse_thought_response(response)
    
    async def _act(self, action: Dict) -> Dict:
        """
        行动阶段：执行工具
        
        Args:
            action: {"tool": "tool_name", "params": {...}}
        """
        tool_name = action.get("tool")
        params = action.get("params", {})
        
        # 查找工具
        tool = self.tools.get(tool_name)
        if not tool:
            return {
                "success": False,
                "error": f"未知工具: {tool_name}"
            }
        
        # 执行工具
        try:
            result = await tool.execute(**params)
            return {
                "success": result.success,
                "data": result.data,
                "error": result.error
            }
        except Exception as e:
            return {
                "success": False,
                "error": str(e)
            }
    
    async def _retry_action(self, action: Dict, last_error: Dict, attempt: int) -> Dict:
        """重试失败的行动"""
        # 可以在这里添加重试逻辑，比如修改参数、换用其他工具等
        return await self._act(action)
    
    def _build_think_prompt(self, context: Dict, history: List[Dict]) -> str:
        """构建思考阶段的 prompt"""
        tools_desc = self.tools.get_tools_description()
        
        prompt = f"""你是一个智能助手，需要根据用户输入和上下文决定下一步行动。

可用工具：
{tools_desc}

用户输入：{context['user_input']}

历史思考：
{self._format_thoughts(context['thoughts'])}

历史行动：
{self._format_actions(context['actions'])}

历史观察：
{self._format_observations(context['observations'])}

请决定下一步：
1. 如果任务已完成，输出最终答案
2. 如果需要使用工具，输出工具调用

请按以下格式响应：
{{
    "thought": "你的思考过程",
    "action": {{"tool": "工具名", "params": {{"参数名": "参数值"}}}},
    "should_stop": false,
    "answer": ""
}}
"""
        return prompt
```

### 2.3 工具注册机制

```python
# app/tools/registry.py
from typing import Dict, Type
from .base import BaseTool

class ToolRegistry:
    """工具注册表"""
    
    _tools: Dict[str, BaseTool] = {}
    
    @classmethod
    def register(cls, tool_instance: BaseTool):
        """注册工具"""
        name = tool_instance.definition.name
        cls._tools[name] = tool_instance
        print(f"[ToolRegistry] 注册工具: {name}")
    
    @classmethod
    def get(cls, name: str) -> BaseTool:
        """获取工具"""
        return cls._tools.get(name)
    
    @classmethod
    def list_tools(cls) -> list:
        """列出所有工具"""
        return list(cls._tools.keys())
    
    @classmethod
    def get_tools_description(cls) -> str:
        """获取所有工具的描述（用于 prompt）"""
        descriptions = []
        for name, tool in cls._tools.items():
            desc = f"- {name}: {tool.definition.description}"
            descriptions.append(desc)
        return "\n".join(descriptions)

# 注册装饰器
def register_tool():
    """工具注册装饰器"""
    def decorator(tool_class: Type[BaseTool]):
        instance = tool_class()
        ToolRegistry.register(instance)
        return tool_class
    return decorator
```

#### 2.4.1 Windows 命令执行规则技术原则

**1. 需求原因**

**背景问题**：
- 原AI Agent Shell（AI_OSShell_v1.0）使用Git Bash执行命令，导致Windows环境不可用
- 实际测试发现目标机器可能未安装Git（特别是普通用户机器）
- 命令执行环境假设不清晰，导致代码在不同环境表现不一致
- 中文路径、UTF-8编码问题频发

**用户价值**：
- 确保AI在纯Windows环境（无Git Bash）也能正常执行文件操作
- 提高命令执行的可靠性，减少环境依赖导致的失败
- 支持中文路径和文件名，提升用户体验

**技术必要性**：
- 文件操作工具（P0）是Phase 1.3的核心功能
- 所有文件操作都依赖命令执行，这是基础中的基础
- 不解决此问题，后续所有工具都无法稳定运行

**2. 核心技术方法**

**总体思路**：不假设目标机器有Git Bash，优先使用Python标准库，其次使用Windows原生命令

**命令选择优先级（经测试验证）**：
```
1. Python 标准库 (首选)  ← 跨平台，无编码问题
2. PowerShell (次选)    ← Windows 原生，功能强大
3. CMD (备选)          ← 兼容性最好，但功能有限
4. 外部工具 (避免)      ← 增加依赖，不稳定
```

**关键设计决策**：
- **禁用Git Bash相关命令**：如 `grep`, `awk`, `ls` 等Linux工具
- **强制UTF-8编码**：所有文件读写必须显式指定 `encoding='utf-8'`
- **使用原始字符串处理路径**：避免Windows路径转义问题
- **优先使用Python标准库**：`pathlib`, `shutil` 替代系统命令

**技术选型理由**：
- Python标准库：跨平台、编码处理好、异常处理完善
- PowerShell：Windows 7+默认安装，功能强大
- 避免CMD：功能有限，编码问题多

#### 2.4.2 日志与可观测性设计原则

**1. 需求原因**

**背景问题**：
- ReAct执行器是多步骤循环（思考→行动→观察），执行过程不透明
- 工具调用失败时难以定位问题（是哪个工具？什么参数？）
- 性能问题无法追踪（哪一步耗时最长？）
- 用户对AI执行过程缺乏信任（不知道做了什么）

**用户价值**：
- 用户可查看AI执行的完整过程，建立信任
- 出现问题时可提供日志给开发者定位
- 可追踪执行性能，优化慢的操作

**技术必要性**：
- ReAct循环复杂，必须有日志才能调试
- 生产环境需要可观测性才能维护
- v2.1规范明确要求日志设计

**2. 核心技术方法**

**总体思路**：建立全链路追踪机制，通过分层日志架构实现不同粒度的可观测性，采用结构化日志格式支持后续分析和可视化。

**实现要点**：首先设计日志分层架构，区分系统级、API级、ReAct级和工具级日志；其次实现request_id的全链路传递机制，确保单次执行的所有日志可关联；最后定义关键日志类型的记录规范，包括执行开始、思考过程、工具调用、重试记录、执行完成和异常错误等场景。

**关键设计决策**：

**1) 日志分层**：
```
系统级日志（app.log）     ← 应用启动、配置、错误
API级日志（api.log）      ← 请求响应、性能指标
ReAct级日志（react.log）  ← 思考过程、工具调用
工具级日志（tool.log）    ← 单个工具执行详情
```

**2) 全链路追踪**：
- 每个ReAct执行生成唯一request_id
- 所有相关日志携带request_id
- 通过request_id可追踪完整执行链

**3) 必须记录的日志类型**：
| 日志类型 | 级别 | 内容 | 用途 |
|---------|------|------|------|
| 执行开始 | INFO | 用户输入、上下文 | 流程追踪 |
| 思考过程 | DEBUG | AI思考、决策依据 | 调试决策逻辑 |
| 工具调用 | INFO | 工具名、参数、耗时 | 监控工具使用 |
| 工具结果 | INFO/WARNING | 成功/失败、返回值 | 追踪执行结果 |
| 重试记录 | WARNING | 重试次数、原因 | 监控稳定性 |
| 执行完成 | INFO | 总耗时、最终答案 | 统计效率 |
| 异常错误 | ERROR | 异常类型、堆栈 | 问题排查 |

**技术选型理由**：
- Python标准logging库：简单、稳定、可扩展
- 结构化JSON日志：便于日志分析和可视化
- 分层设计：不同场景查看不同日志，避免信息过载

#### 2.4.3 文件操作安全设计原则

**1. 需求原因**

**背景问题**：
- AI执行文件操作具有破坏性（删除、移动可能导致数据丢失）
- 用户需要知道AI做了什么操作（可审计性）
- 操作出错时需要能恢复到之前状态（可回退性）
- 误删文件后需要能恢复（数据保护）

**用户价值**：
- 建立用户对AI的信任（知道AI做了什么，可以撤销）
- 误操作后可恢复，降低使用风险
- 可查看完整操作历史，满足审计需求

**技术必要性**：
- 文件操作是高风险操作，必须有安全机制
- v2.1规范要求"闭环整改"，安全设计是闭环的基础
- 没有安全机制，产品无法在生产环境使用

**2. 核心技术方法**

**总体思路**：通过建立完整的操作记录体系和备份机制，实现文件操作的可追溯、可回退、可恢复三重安全保障，确保AI文件操作的可靠性和用户数据的安全性。

**实现要点**：首先建立操作历史记录机制，使用SQLite数据库存储每次操作的完整信息；其次实现删除文件备份机制，将删除的文件移动到回收站并保留30天；再次建立移动文件映射表，记录文件移动的源/目标路径关系；最后实现撤销/回退功能，支持按逆序回滚单个操作或整个会话的所有操作。

**四大安全机制**：

**1) 操作历史记录（可追溯）**：
- 记录每次文件操作的完整信息（类型、路径、时间、结果）

**2) 删除文件备份（可恢复）**：

**3) 移动文件映射（可追踪）**：
- 记录文件移动的源路径和目标路径
- 支持查看文件原来在哪里
- 支持回滚到原始位置

**4) 撤销/回退功能（可回退）**：
- 支持撤销单个操作
- 支持回滚整个会话的所有操作
- 按逆序执行回滚，确保一致性

**技术选型理由**：
- SQLite数据库：轻量、可靠、无需额外服务
- 文件系统备份：简单、通用、不依赖外部存储
- 事务性记录：确保操作和记录的一致性

```python
from app.utils.logger import api_logger
import time

class ReActExecutor:
    """ReAct 执行器（含日志记录）"""
    
    async def execute(self, user_input: str, history: List[Dict] = None) -> Dict:
        # 生成请求 ID 用于全链路追踪
        request_id = api_logger.log_request_start(
            provider="react_executor",
            model="react_loop",
            message_len=len(user_input),
            history_count=len(history) if history else 0
        )
        
        start_time = time.time()
        iterations = 0
        
        try:
            while iterations < self.max_iterations:
                iterations += 1
                
                # 记录思考过程
                self.logger.debug(f"[{request_id}] 第{iterations}轮思考开始")
                thought = await self._think(context, history)
                self.logger.debug(f"[{request_id}] 思考: {thought.get('thought', '')[:100]}...")
                
                # 检查是否结束
                if thought.get("should_stop", False):
                    self.logger.info(f"[{request_id}] 任务完成，共{iterations}轮")
                    break
                
                # 记录工具调用
                action = thought.get("action")
                if action:
                    tool_name = action.get("tool")
                    self.logger.info(f"[{request_id}] 调用工具: {tool_name}")
                    
                    # 执行工具
                    tool_start = time.time()
                    observation = await self._act(action)
                    tool_elapsed = time.time() - tool_start
                    
                    # 记录工具结果
                    if observation.get("success"):
                        self.logger.info(
                            f"[{request_id}] 工具执行成功: {tool_name}, "
                            f"耗时: {tool_elapsed:.2f}s"
                        )
                    else:
                        self.logger.warning(
                            f"[{request_id}] 工具执行失败: {tool_name}, "
                            f"错误: {observation.get('error', 'Unknown')}"
                        )
                        
                        # 记录重试
                        for attempt in range(self.retry_attempts):
                            self.logger.info(f"[{request_id}] 第{attempt+1}次重试...")
                            # ... 重试逻辑
            
            # 记录执行完成
            total_elapsed = time.time() - start_time
            api_logger.log_response_with_time(
                request_id=request_id,
                provider="react_executor",
                status_code=200 if context.get("success") else 500,
                content_len=len(context.get("final_answer", "")),
                error=None if context.get("success") else context.get("error")
            )
            
            self.logger.info(
                f"[{request_id}] ReAct执行完成, "
                f"总耗时: {total_elapsed:.2f}s, "
                f"迭代次数: {iterations}"
            )
            
        except Exception as e:
            self.logger.exception(f"[{request_id}] ReAct执行异常")
            api_logger.log_error("react_executor", str(e))
            raise
```

#### 3.3.2 删除文件备份机制（回收站）

**问题**: 用户说"整理下载文件夹"，AI 删除了重要文件，用户想恢复

**解决方案**: 删除的文件先移动到备份目录（类似回收站）

```python
# app/utils/recycle_bin.py
from pathlib import Path
import shutil
from datetime import datetime

class RecycleBin:
    """回收站 - 删除文件备份机制"""
    
    def __init__(self, base_path: str = "data/recycle_bin"):
        self.base_path = Path(base_path)
        self.base_path.mkdir(parents=True, exist_ok=True)
        
        # 按日期组织备份
        self.current_session = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.session_path = self.base_path / self.current_session
        self.session_path.mkdir(exist_ok=True)
    
    def backup_before_delete(self, original_path: str, operation_id: str) -> str:
        """
        删除前备份文件
        
        Args:
            original_path: 原文件路径
            operation_id: 操作ID（用于后续恢复）
            
        Returns:
            backup_path: 备份路径
        """
        original = Path(original_path)
        if not original.exists():
            raise FileNotFoundError(f"文件不存在: {original_path}")
        
        # 创建备份目录结构
        backup_dir = self.session_path / operation_id
        backup_dir.mkdir(parents=True, exist_ok=True)
        
        # 备份文件
        backup_file = backup_dir / original.name
        if original.is_file():
            shutil.copy2(original, backup_file)
        elif original.is_dir():
            shutil.copytree(original, backup_file)
        
        # 保存元数据
        metadata = {
            "operation_id": operation_id,
            "original_path": str(original.absolute()),
            "backup_path": str(backup_file),
            "timestamp": datetime.now().isoformat(),
            "file_size": original.stat().st_size if original.is_file() else 0,
            "is_directory": original.is_dir()
        }
        
        # 保存元数据文件
        metadata_file = backup_dir / "metadata.json"
        import json
        with open(metadata_file, 'w', encoding='utf-8') as f:
            json.dump(metadata, f, ensure_ascii=False, indent=2)
        
        return str(backup_file)
    
    def restore(self, operation_id: str) -> str:
        """
        恢复删除的文件
        
        Args:
            operation_id: 操作ID
            
        Returns:
            restored_path: 恢复后的路径
        """
        backup_dir = self.session_path / operation_id
        metadata_file = backup_dir / "metadata.json"
        
        if not metadata_file.exists():
            raise FileNotFoundError(f"备份不存在: {operation_id}")
        
        # 读取元数据
        import json
        with open(metadata_file, 'r', encoding='utf-8') as f:
            metadata = json.load(f)
        
        original_path = Path(metadata["original_path"])
        backup_path = Path(metadata["backup_path"])
        
        # 恢复到原位置（如果原位置已被占用，恢复到原位置_恢复时间）
        if original_path.exists():
            timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
            original_path = original_path.parent / f"{original_path.name}_ restored_{timestamp}"
        
        # 恢复文件/目录
        if backup_path.is_file():
            shutil.copy2(backup_path, original_path)
        elif backup_path.is_dir():
            shutil.copytree(backup_path, original_path)
        
        return str(original_path)
    
    def list_backups(self) -> List[dict]:
        """列出所有可恢复的备份"""
        backups = []
        for session_dir in self.base_path.iterdir():
            if session_dir.is_dir():
                for backup_dir in session_dir.iterdir():
                    metadata_file = backup_dir / "metadata.json"
                    if metadata_file.exists():
                        import json
                        with open(metadata_file, 'r', encoding='utf-8') as f:
                            metadata = json.load(f)
                        backups.append(metadata)
        return backups
    
    def cleanup_old_backups(self, days: int = 30):
        """清理超过指定天数的备份"""
        from datetime import timedelta
        cutoff = datetime.now() - timedelta(days=days)
        
        for session_dir in self.base_path.iterdir():
            if session_dir.is_dir():
                try:
                    session_time = datetime.strptime(session_dir.name, "%Y%m%d_%H%M%S")
                    if session_time < cutoff:
                        shutil.rmtree(session_dir)
                except ValueError:
                    continue
```

```python
from typing import List, Optional
from datetime import datetime

class RollbackManager:
    """回退管理器"""
    
    def __init__(self):
        self.operation_history = OperationHistoryStore()
        self.recycle_bin = RecycleBin()
        self.file_mapping = FileMappingRegistry()
    
    async def rollback_single_operation(self, operation_id: str) -> dict:
        """
        回退单个操作
        
        Args:
            operation_id: 操作ID
            
        Returns:
            回退结果
        """
        operation = self.operation_history.get_operation(operation_id)
        
        if not operation:
            return {"success": False, "error": "操作不存在"}
        
        if not operation.can_rollback:
            return {"success": False, "error": "该操作不可回退"}
        
        if operation.operation_type == OperationType.DELETE:

            restored_path = self.recycle_bin.restore(operation_id)
            self.operation_history.mark_rollbacked(operation_id)

                "success": True,
                "operation": "delete",
                "action": "restore",
                "restored_path": restored_path
            }
        
        elif operation.operation_type == OperationType.MOVE:
            # 反向移动
            success = self.file_mapping.rollback_move(operation_id)
            if success:
                self.operation_history.mark_rollbacked(operation_id)
                return {
                    "success": True,
                    "operation": "move",
                    "action": "rollback",
                    "message": "文件已移回原位置"
                }
            else:
                return {"success": False, "error": "回退失败，文件可能已被修改"}
        
        elif operation.operation_type == OperationType.WRITE:
            # 写入操作回退比较复杂，需要保存原始内容
            # ... 实现略
            pass
        
        else:
            return {"success": False, "error": f"不支持回退的操作类型: {operation.operation_type}"}
    
    async def rollback_execution(self, execution_id: str) -> dict:
        """
        回退整个 ReAct 执行会话（逆序回退所有操作）
        
        Args:
            execution_id: 执行会话ID
        """
        operations = self.operation_history.get_operations_by_execution(execution_id)
        
        # 按时间倒序回退（后做的先回退）
        results = []
        for op in reversed(operations):
            if op.can_rollback and op.status != OperationStatus.ROLLED_BACK:
                result = await self.rollback_single_operation(op.id)
                results.append(result)
        
        return {
            "success": True,
            "execution_id": execution_id,
            "total_operations": len(operations),
            "rollbacked_count": len([r for r in results if r["success"]]),
            "details": results
        }
```

| `executed_at` | datetime | 操作执行时间 | ✅ 已有 |

| 字段 | 类型 | 用途 | 建议添加 |
|------|------|------|----------|
| `file_extension` | string | 文件扩展名（用于按类型统计） | ⚠️ 建议添加 |

**数据生成策略**：
- `file_extension`: 从 `source_path` 或 `destination_path` 解析提取
- `duration_ms`: 记录 `executed_at - created_at` 的毫秒差
- `space_impact_bytes`: 
  - delete: +file_size (释放空间)
  - create: -file_size (占用空间)
  - move/copy: 0 (无净变化)

##### 后端数据服务实现代码示例

> **职责说明**: 后端只负责生成**数据**（文本/JSON/Mermaid语法），不负责**渲染**。渲染由前端使用专业可视化库完成。

```python
# app/services/visualization.py
# 后端职责：数据服务层（生成数据，不处理渲染）

class OperationVisualizer:
    """
    操作过程可视化数据服务
    
    职责边界：
    - ✅ 生成文本格式报告（.txt）- 纯文本无需渲染
    - ✅ 生成JSON数据结构 - 供前端使用
    - ✅ 生成Mermaid语法 - 供Mermaid引擎渲染
    - ✅ 生成独立HTML报告 - 内嵌JS用于下载分享
    - ❌ 不生成可视化图表（D3/ECharts）- 这是前端职责
    """
    
    def generate_text_report(self, execution_id: str) -> str:
        """
        生成文本格式报告（后端职责）
        
        用途: 快速查看、命令行输出、日志记录
        特点: 纯文本，无需渲染引擎
        """
        operations = self._get_operations(execution_id)
        
        lines = ["📁 文件整理过程可视化", "=" * 50]
        
        # 按操作类型分组
        moves = [op for op in operations if op.type == "move"]
        deletes = [op for op in operations if op.type == "delete"]
        
        lines.append("\n📦 文件移动:\n")
        for op in moves:
            lines.append(f"  {op.source}")
            lines.append(f"  {' ' * len(op.source)} └─➡️ {op.destination}")
        
        lines.append("\n🗑️  删除的文件:\n")
        for op in deletes:
            lines.append(f"  ❌ {op.source} (可恢复)")
        
        lines.append("\n" + "=" * 50)
        lines.append(f"📊 总计: 移动 {len(moves)} 个, 删除 {len(deletes)} 个")
        
        return "\n".join(lines)
    
    def generate_tree_data(self, execution_id: str) -> dict:
        """
        生成树形结构数据（后端职责）
        
        用途: 供前端 D3.js/ECharts 渲染交互式树图
        返回: 标准JSON树形结构
        """
        operations = self._get_operations(execution_id)
        
        # 构建标准树形结构数据（不渲染）
        tree_data = {
            "id": execution_id,
            "name": "文件整理任务",
            "type": "root",
            "children": []
        }
        
        for op in operations:
            node = {
                "id": op.operation_id,
                "name": op.filename or "unknown",
                "type": op.operation_type,
                "source": op.source_path,
                "destination": op.destination_path,
                "status": op.status,
                "size": op.file_size,
                "timestamp": op.executed_at.isoformat() if op.executed_at else None
            }
            tree_data["children"].append(node)
        
        return tree_data  # 前端使用 D3.js/ECharts 渲染
    
    def generate_mermaid_syntax(self, execution_id: str) -> str:
        """
        生成 Mermaid 流程图语法（后端职责）
        
        用途: 生成.mmd文件，可用Mermaid CLI或在线编辑器渲染
        特点: 文本格式，Mermaid引擎负责渲染（非后端）
        """
        operations = self._get_operations(execution_id)
        
        mermaid = ["graph LR"]
        
        for op in operations:
            if op.type == "move":
                source = self._sanitize(op.source)
                dest = self._sanitize(op.destination)
                mermaid.append(f"    {source}[{op.filename}] -->|移动| {dest}")
            elif op.type == "delete":
                source = self._sanitize(op.source)
                mermaid.append(f"    {source}[{op.filename}] -->|删除| trash[🗑️]")
        
        return "\n".join(mermaid)  # Mermaid引擎渲染，非后端渲染
    
    def generate_stats_data(self, execution_id: str) -> dict:
        """
        生成统计图表数据（后端职责）
        
        用途: 供前端 ECharts/Chart.js 渲染统计图表
        返回: 标准图表数据结构（ECharts格式兼容）
        """
        operations = self._get_operations(execution_id)
        
        # 按操作类型统计
        type_counts = {}
        for op in operations:
            type_counts[op.type] = type_counts.get(op.type, 0) + 1
        
        # 按文件扩展名统计
        ext_counts = {}
        for op in operations:
            ext = op.file_extension or "unknown"
            ext_counts[ext] = ext_counts.get(ext, 0) + 1
        
        # 空间变化计算
        space_freed = sum([o.file_size for o in operations if o.type == "delete"])
        
        return {
            "operation_types": {
                "title": "操作类型分布",
                "data": [
                    {"name": name, "value": count}
                    for name, count in type_counts.items()
                ]
            },
            "file_extensions": {
                "title": "文件类型分布",
                "data": [
                    {"name": ext, "value": count}
                    for ext, count in ext_counts.items()
                ]
            },
            "space_change": {
                "title": "空间变化",
                "freed_bytes": space_freed,
                "freed_formatted": self._format_size(space_freed)
            },
            "timeline": [
                {
                    "timestamp": op.executed_at.isoformat() if op.executed_at else None,
                    "type": op.type,
                    "file": op.filename
                }
                for op in sorted(operations, key=lambda x: x.sequence_number)
            ]
        }  # 前端使用 ECharts 渲染图表
    
    def generate_html_report(self, execution_id: str, output_path: str) -> str:
        """
        生成独立HTML报告（后端职责）
        
        用途: 可下载的完整报告（含内嵌JS，无需后端）
        特点: 一次性生成，独立运行，用于分享导出
        注意: 虽然含JS，但这是静态文件生成，非后端实时渲染
        """
        # 获取所有数据
        stats = self.generate_stats_data(execution_id)
        tree_data = self.generate_tree_data(execution_id)
        
        # 生成包含内嵌JS的HTML（用于下载分享）
        html_content = f"""<!DOCTYPE html>
```

**可视化职责边界总结表**:

| 可视化类型 | 后端职责 | 前端职责 | 技术栈 | 说明 |
|-----------|---------|---------|--------|------|
| **文本列表** | 生成文本格式 | 直接显示 | Python字符串 | 纯文本，无需渲染 |
| **ASCII树形** | 生成ASCII文本 | 直接显示（pre标签） | Python字符串 | 文本格式，终端友好 |
| **JSON树形数据** | 生成树形结构JSON | D3.js/ECharts渲染 | D3.js/ECharts | 后端只提供数据 |
| **Mermaid语法** | 生成.mmd文本 | Mermaid引擎渲染 | Mermaid.js | 后端生成语法 |
| **Sankey流向数据** | 生成source/target数组 | D3 Sankey渲染 | D3.js | 后端只提供数据 |
| **统计数据** | 生成统计值 | ECharts渲染 | ECharts | 后端计算，前端绘图 |
| **动画时间轴** | 生成时间轴数据 | CSS/JS动画渲染 | Web Animations | 后端只提供数据 |
| **独立HTML报告** | 生成含内嵌JS的HTML | 浏览器渲染 | ECharts CDN | 一次性静态文件 |
| **3D可视化** | ❌ 不提供 | Three.js渲染 | Three.js/WebGL | Phase 2.x前端功能 |

**关键原则**:
1. **后端不渲染图表**: 后端只生成数据和简单文本，不处理任何图表渲染
2. **前端负责展示**: 所有交互式可视化（树形图、Sankey、统计图、动画）由前端使用专业库渲染
3. **独立HTML例外**: 虽然HTML含JS，但这是一次性生成的静态文件，用于下载分享，非后端实时服务
4. **数据格式标准**: 后端生成的JSON使用标准格式（兼容ECharts/D3.js），便于前端直接使用

- 中文路径需验证：包含中文的文件名和目录名都能正确处理

**文件操作安全验收标准**（v1.3 新增）：
- **删除备份**：删除的文件必须在 `data/recycle_bin/` 目录有备份，保留30天
- **移动记录**：必须记录源路径→目标路径的映射关系，支持链式追踪（A→B→C→当前位置）

- **撤销功能**：

### 6.1 技术风险

| 操作历史丢失 | 低 | 高 | SQLite 数据库定期备份 |

**风险补充说明**（v1.2 新增）：
- **AI 响应格式不稳定**：ReAct 依赖 AI 输出标准 JSON，可能解析失败，需添加 try-except 和重试
- **日志系统故障**：如果 Phase 1.2 的日志系统有问题，会影响 ReAct 调试，需确保 api_logger 初始化正确

**文件操作安全风险**（v1.3 新增）：
- **回收站空间不足**：大量删除操作可能占满磁盘，需监控并自动清理30天前的备份
- **回退冲突**：用户回退时，如果文件已被手动修改，应提示冲突而不是强制覆盖
- **操作历史丢失**：SQLite 数据库损坏可能导致历史记录丢失，需定期备份数据库文件
- **并发操作冲突**：用户和 AI 同时操作同一文件，需文件锁或乐观锁机制

**响应体**:

```json
      "action": {"tool": "read_file", "params": {"path": "file.txt"}},
```

#### 8.3.2 现有端点修改

**端点 2: 标准对话端点（增强）**
```
POST /api/v1/chat
```

**新增参数**:
```json
{
  "message": "用户输入",
  "use_react": true,  // 新增参数：是否使用 ReAct 模式
  "history": []
}
```

**行为**:
- `use_react: false`（默认）：直接调用 AI 服务，简单对话
- `use_react: true`：使用 ReAct 执行器，支持工具调用

#### 8.3.3 API 实现代码示例

```python
# app/api/v1/chat.py
from app.core.react_executor import ReActExecutor
from app.tools.registry import ToolRegistry

@router.post("/chat/react")
async def chat_with_react(
    request: ChatRequest,
    background_tasks: BackgroundTasks
) -> ChatResponse:
    """使用 ReAct 执行对话"""
    # 获取 AI 服务
    ai_service = AIServiceFactory.get_service()
    
    # 获取工具注册表
    tool_registry = ToolRegistry()
    
    # 创建 ReAct 执行器
    executor = ReActExecutor(ai_service, tool_registry)
    
    # 执行 ReAct 循环
    result = await executor.execute(
        user_input=request.message,
        history=request.history
    )
    
    return ChatResponse(
        success=result["success"],
        answer=result.get("answer", ""),
        execution_log=result.get("context", {}),
        total_iterations=len(result.get("context", {}).get("actions", [])),
        total_time=result.get("total_time", 0)
    )
```
