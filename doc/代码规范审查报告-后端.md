```python
def _validate_dependencies(self, dependencies):
```

```python
"""conversation_history管理"""
from typing import List, Dict, Any

class HistoryManager:
    """conversation_history管理"""
    
    def __init__(self, max_context_chars: int):
        self.max_context_chars = max_context_chars
        self.conversation_history: List[Dict[str, Any]] = []
        self.temp_history: List[Dict[str, Any]] = []
    
    def init_history(self, sys_prompt: str, task_prompt: str):
        """初始化历史"""
        self.conversation_history = [
            {"role": "system", "content": sys_prompt},
            {"role": "user", "content": task_prompt},
        ]
        self.temp_history = []
    
    def add_assistant(self, content: str):
        """添加assistant消息"""
        self.temp_history.append({"role": "assistant", "content": content})
    
    def add_observation(self, observation_text: str):
        """添加observation消息"""
        self.temp_history.append({"role": "user", "content": observation_text})
    
    def add_parse_error(self, error_msg: str):
        """添加解析错误消息"""
        self.temp_history.append({"role": "user", "content": f"[解析错误] {error_msg}"})
    
    def flush_temp_to_history(self, chunk_buffer=None):
        """将temp_history刷新到conversation_history"""
        if chunk_buffer and chunk_buffer.buffer:
            combined = "".join(chunk_buffer.buffer)
            if combined.strip():
                self.temp_history.append({"role": "assistant", "content": combined})
        
        self.conversation_history.extend(self.temp_history)
        self.temp_history = []
    
    def prepare_messages_for_llm(self) -> List[Dict[str, Any]]:
        """准备发送给LLM的消息"""
        return self.conversation_history + self.temp_history
    
    def reset_per_run(self):
        """重置每次运行的状态"""
        self.temp_history = []
```
