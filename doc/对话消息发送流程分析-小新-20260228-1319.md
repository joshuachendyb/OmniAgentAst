## 一、文档概述

本文档详细分析了用户在前端对话框输入消息（如"你好"）并按回车后，整个系统的代码执行流程和逻辑。涵盖从前端输入到后端处理的完整链路。

---

## 二、流程总览图

```
用户输入"你好" + 回车
        ↓
【前端】ChatContainer.tsx
    onPressEnter → handleSend()
        ↓
【前端】sse.ts
    sendStreamMessage('你好')
    POST → http://localhost:8000/api/v1/chat/stream
        ↓
【后端】chat.py
    chat_stream() 接收请求
    → 发送 start 事件
    → 发送 thought 事件（"正在分析任务..."）
    → 调用 AI 服务（opencode/minimax-m2.5-free）
    → 流式返回 chunk 事件（逐字）
    → 发送 final 事件（完成）
        ↓
【前端】sse.ts
    processSSEData() 解析SSE数据
    → chunk: 逐字更新AI消息
    → final: 完成消息，更新状态
        ↓
【界面】显示AI回复内容
```

```tsx
const handleSend = async () => {
  // 1. 检查输入是否为空或正在加载
  if (!inputValue.trim() || loading) return;

  // 2. 获取用户输入内容
  const userContent = inputValue.trim();  // "你好"
  
  // 3. 清空输入框
  setInputValue('');
  
  // 4. 设置加载状态
  setLoading(true);

  // 5. 添加用户消息到消息列表（立即显示在界面）
  const userMessage: Message = {
    id: Date.now().toString(),
    role: 'user',
    content: userContent,  // "你好"
    timestamp: new Date(),
  };
  setMessages((prev) => [...prev, userMessage]);

  // 6. 流式模式处理
  if (useStream) {
    // 6.1 添加AI消息占位（空的，等待填充）
    const assistantMessage: Message = {
      id: (Date.now() + 1).toString(),
      role: 'assistant',
      content: '',
      timestamp: new Date(),
      executionSteps: [],
      isStreaming: true,
      model: serviceStatus?.model || currentModel || 'unknown',
    };
    setMessages((prev) => [...prev, assistantMessage]);

    // 6.2 清空之前的执行步骤
    clearSteps();

    // 6.3 发送流式请求
    sendStreamMessage(userContent);  // ← 核心：发送"你好"
  }
};
```

**说明**：
- 获取用户输入的"你好"
- 立即在界面上显示用户消息
- 创建一个空的AI消息占位符
- 调用 `sendStreamMessage('你好')` 发送请求

### 3.3 第三步：sendStreamMessage 函数

**文件位置**: `frontend/src/utils/sse.ts`  
**代码位置**: 第163-285行

```tsx
const sendMessage = useCallback(
  async (content: string) => {
    console.log('[SSE sendMessage] 函数被调用, content:', content);
    
    // 1. 断开已有连接，清空步骤
    disconnect();
    clearSteps();

    // 2. 设置接收状态
    setIsReceiving(true);
    setIsConnected(true);

    try {
      // 3. 构建请求URL
      const url = `${config.baseURL}/chat/stream`;
      // 实际URL: http://localhost:8000/api/v1/chat/stream

      // 4. 创建超时控制器
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 60000); // 60秒超时

      // 5. 发送POST请求
      const response = await fetch(url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          ...(config.token ? { Authorization: `Bearer ${config.token}` } : {}),
        },
        body: JSON.stringify({
          messages: [{ role: 'user', content: content }],  // [{ role: 'user', content: '你好' }]
          stream: true,
          task_id: taskId || undefined,
        }),
        signal: controller.signal,
      });

      clearTimeout(timeoutId);

      // 6. 检查响应状态
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }

      if (!response.body) {
        throw new Error('响应体为空');
      }

      // 7. 使用ReadableStream读取流式数据
      const reader = response.body.getReader();
      const decoder = new TextDecoder('utf-8');
      let buffer = '';

      console.log('[SSE] 开始接收流式数据...');

      // 8. 循环读取数据
      while (true) {
        const { done, value } = await reader.read();

        if (done) {
          // 处理剩余的buffer数据
          if (buffer.trim()) {
            processSSEData(buffer, handlers, isProcessingRef);
          }
          console.log('[SSE] 流式接收完成');
          break;
        }

        // 解码数据块
        buffer += decoder.decode(value, { stream: true });

        // 按行分割，处理SSE格式
        const lines = buffer.split('\n');
        buffer = lines.pop() || ''; // 保留最后一个不完整的行

        for (const line of lines) {
          processSSEData(line, handlers, isProcessingRef);
        }
      }
    } catch (error: any) {
      console.error('[SSE] 请求错误:', error);
      setIsConnected(false);
      setIsReceiving(false);

      if (error.name === 'AbortError') {
        message.warning('请求超时');
        onError?.('请求超时');
      } else {
        message.error(`连接异常: ${error.message}`);
        onError?.(error.message);
      }
    }
  },
  [config, disconnect, clearSteps, onStep, onChunk, onComplete, onError]
);
```

**说明**：
- 构建请求URL: `http://localhost:8000/api/v1/chat/stream`
- 发送POST请求，body包含用户消息
- 使用 ReadableStream 逐行读取SSE数据
- 每行数据调用 `processSSEData()` 处理

```python
@router.post("/chat/stream")
async def chat_stream(request: ChatRequest):
    """
    流式API - SSE (Server-Sent Events)
    
    实时推送ReAct执行步骤：思考→行动→观察
    支持任务中断（前端使用AbortController）
    """
    import uuid
    
    # 1. 生成或获取任务ID
    task_id = request.task_id if request.task_id else str(uuid.uuid4())
    
    async def generate():
        """生成SSE流，支持中断"""
        # 2. 注册任务
        async with running_tasks_lock:
            running_tasks[task_id] = {
                "status": "running", 
                "cancelled": False,
                "created_at": datetime.now()
            }
        
        # 3. 获取AI服务实例
        if request.provider and request.model:
            ai_service = AIServiceFactory.get_service_for_model(
                request.provider, 
                request.model
            )
        else:
            ai_service = AIServiceFactory.get_service()
        
        # 4. 发送start事件
        display_name = f"{PROVIDER_DISPLAY_NAMES.get(ai_service.provider, ai_service.provider)} ({ai_service.model})"
        yield f"data: {json.dumps({
            'type': 'start',
            'display_name': display_name,
            'provider': ai_service.provider,
            'model': ai_service.model,
            'task_id': task_id
        })}\n\n"
        
        try:
            # 5. 获取最后一条用户消息
            last_message = request.messages[-1].content  # "你好"
            
            # 6. 发送思考事件
            yield f"data: {json.dumps({'type': 'thought', 'content': '正在分析任务...'})}\n\n"
            await asyncio.sleep(0.3)
            
            # 7. 检查是否被中断
            async with running_tasks_lock:
                if running_tasks.get(task_id, {}).get("cancelled", False):
                    yield f"data: {json.dumps({'type': 'interrupted', 'content': '任务已被中断'})}\n\n"
                    return
            
            # 8. 检测文件操作意图
            is_file_op, _, confidence = detect_file_operation_intent(last_message)
            
            if is_file_op and confidence >= 0.3:
                # 文件操作处理逻辑...
                pass
            else:
                # 9. 普通对话：调用AI服务
                yield f"data: {json.dumps({'type': 'action', 'step': 1, 'content': '正在调用AI服务...'})}\n\n"
                
                ai_service = AIServiceFactory.get_service()
                from app.services.base import Message
                history = []
                if len(request.messages) > 1:
                    for msg in request.messages[:-1]:
                        history.append(Message(role=msg.role, content=msg.content))
                
                # 10. 流式调用AI，逐token返回
                full_content = ""
                async for chunk in ai_service.chat_stream(
                    message=last_message,  # "你好"
                    history=history
                ):
                    # 检查是否被中断
                    async with running_tasks_lock:
                        if running_tasks.get(task_id, {}).get("cancelled", False):
                            yield f"data: {json.dumps({'type': 'interrupted', 'content': '任务已被中断'})}\n\n"
                            return
                    
                    if chunk.content:
                        full_content += chunk.content
                        # 逐token发送到前端
                        yield f"data: {json.dumps({
                            'type': 'chunk', 
                            'content': chunk.content, 
                            'model': chunk.model, 
                            'provider': ai_service.provider
                        })}\n\n"
                    
                    if chunk.is_done:
                        break
                
                # 11. 发送最终结果
                yield f"data: {json.dumps({
                    'type': 'final', 
                    'content': full_content, 
                    'model': ai_service.model, 
                    'provider': ai_service.provider
                })}\n\n"
                        
        except asyncio.CancelledError:
            # 客户端断开连接
            async with running_tasks_lock:
                running_tasks[task_id] = {"status": "cancelled", "cancelled": True}
            yield f"data: {json.dumps({'type': 'interrupted', 'content': '客户端断开连接，任务中断'})}\n\n"
            
        except Exception as e:
            yield f"data: {json.dumps({'type': 'error', 'content': f'流式响应出错: {str(e)}'})}\n\n"
        
        finally:
            # 清理任务
            async with running_tasks_lock:
                if task_id in running_tasks:
                    del running_tasks[task_id]
    
    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no"
        }
    )
```

**说明**：
- 接收POST请求 `/api/v1/chat/stream`
- 发送 `start` 事件（包含provider、model、task_id）
- 发送 `thought` 事件（"正在分析任务..."）
- 检测是否文件操作意图
- 调用AI服务流式接口，逐token返回
- 发送 `final` 事件（完成）

### 5.1 第五步：processSSEData 函数

**文件位置**: `frontend/src/utils/sse.ts`

```tsx
const processSSEData = (
  line: string,
  handlers: {...},
  isProcessingRef: React.MutableRefObject<boolean>
) => {
  // 跳过空行
  if (!line.trim() || !line.startsWith('data: ')) {
    return;
  }

  try {
    // 解析SSE数据（去掉 "data: " 前缀）
    const rawData = JSON.parse(line.slice(6));

    console.log('[SSE] 收到数据:', rawData.type, rawData);

    // 构建ExecutionStep对象
    const step: ExecutionStep = {
      type: rawData.type as ExecutionStep['type'],
      content: rawData.content || rawData.error || '',
      tool: rawData.action,
      params: rawData.action_input,
      result: rawData.observation,
      timestamp: Date.now(),
      stepNumber: rawData.step || 1,
    };

    switch (rawData.type) {
      case 'start':
        // 收到start事件，保存后端返回的task_id
        console.log('[SSE] 收到start事件:', rawData);
        if (rawData.task_id && setServerTaskId) {
          setServerTaskId(rawData.task_id);
        }
        break;

      case 'thought':
        // 思考步骤
        console.log('[SSE] 思考:', step.content);
        setExecutionSteps((prev) => [...prev, step]);
        onStep?.(step);
        break;

      case 'action':
        // 行动步骤
        console.log('[SSE] 行动:', step.content);
        setExecutionSteps((prev) => [...prev, step]);
        onStep?.(step);
        break;

      case 'observation':
        // 观察结果
        console.log('[SSE] 观察:', step.result);
        setExecutionSteps((prev) => [...prev, step]);
        onStep?.(step);
        break;

      case 'chunk':
        // 流式内容片段 - 逐token返回
        console.log('[SSE] 内容片段:', rawData.content);
        responseBufferRef.current += rawData.content || '';
        setCurrentResponse(responseBufferRef.current);
        onChunk?.(rawData.content || '');  // ← 更新界面
        break;

      case 'final':
        // 最终结果
        console.log('[SSE] 最终结果:', step.content);
        if (step.content && !responseBufferRef.current) {
          responseBufferRef.current = step.content;
          setCurrentResponse(responseBufferRef.current);
          onChunk?.(step.content);
        }
        const model = rawData.model;
        setIsReceiving(false);
        setIsConnected(false);
        disconnect();
        if (onComplete) {
          onComplete(responseBufferRef.current, model);  // ← 完成回调
        }
        break;

      case 'interrupted':
        // 中断
        setIsReceiving(false);
        setIsConnected(false);
        disconnect();
        message.warning(step.content || '任务被中断');
        onError?.(step.content || '任务被中断');
        break;

      case 'error':
        // 错误
        setIsReceiving(false);
        setIsConnected(false);
        disconnect();
        message.error(`流式传输错误: ${step.content}`);
        onError?.(step.content || '未知错误');
        break;

      default:
        console.log('[SSE] 未知类型:', rawData.type);
    }
  } catch (err) {
    console.error('[SSE] 解析数据失败:', err);
  }
};
```

**说明**：
- 解析SSE格式的数据行
- 根据事件类型分别处理：
  - `start`: 保存task_id
  - `thought`: 显示思考过程
  - `chunk`: 逐字追加到AI消息
  - `final`: 完成消息

## 六、SSE事件类型说明

| 事件类型 | 说明 | 触发时机 |
|---------|------|---------|
| `start` | 开始事件 | 连接建立后立即发送 |
| `thought` | 思考步骤 | AI正在分析任务 |
| `action` | 行动步骤 | AI执行某个操作 |
| `observation` | 观察结果 | 操作执行结果 |
| `chunk` | 内容片段 | 逐token返回AI回复 |
| `final` | 最终结果 | AI回复完成 |
| `error` | 错误 | 发生错误时 |
| `interrupted` | 中断 | 任务被中断时 |
