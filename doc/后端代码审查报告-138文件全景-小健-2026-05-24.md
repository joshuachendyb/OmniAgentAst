**组件1: `ChunkBuffer`类**（独立文件 `agent/chunk_buffer.py`）

```python
class ChunkBuffer:
    """管理chunk拼接、阈值检测、flush到message_builder。消除3处重复flush逻辑"""
    def __init__(self, max_consecutive: int = 10):
        self.buffer: str = ""
        self.consecutive_count: int = 0
        self.max_consecutive: int = max_consecutive

    def append(self, content: str) -> None:
        self.buffer += content
        self.consecutive_count += 1

    def should_promote(self) -> bool:
        return self.consecutive_count >= self.max_consecutive

    def flush_to(self, builder: "MessageBuilder") -> str:
        result = self.buffer
        if result:
            builder.temp_history.clear()
            builder.add_assistant(result)
        self.clear()
        return result

    def clear(self) -> None:
        self.buffer = ""
        self.consecutive_count = 0
```

**组件2: 重构后的`run_stream`**（保留在`BaseAgent`中）

```python
async def run_stream(
    self,
    task: Any,
    context: Optional[Dict[str, Any]] = None,
    max_steps: int = 20,
    task_id: Optional[str] = None,
) -> AsyncGenerator[Union[Step, SSEEvent], None]:
    chunk_buffer, valid_tool_names = self._initialize_run_state(task, task_id)
    step_count = 0

    while step_count < max_steps:
        if await self._should_interrupt_and_yield(step_count, task_id):
            return

        try:
            response = await self._get_llm_response(task)
        except Exception as e:
            yield self._handle_run_exception(e, step_count)
            return

        if self._is_empty_response(response):
            if not self._handle_empty_response(len(self.conversation_history)):
                yield self._create_fatal_error_step("max_retries_exceeded")
                return
            continue

        parsed = self._parse_llm_response(response)
        handler = HANDLER_MAP.get(parsed["type"])
        if handler is None:
            logger.error(f"未知parsed type: {parsed['type']}")
            continue

        async for step in handler(self, parsed, step_count, chunk_buffer, valid_tool_names):
            yield step
            if isinstance(step, (FinalStep, ErrorStep)):
                return
        step_count += 1
```

```python
def _is_empty_response(self, response: Any) -> bool:
    if response is None:
        return True
    content = getattr(response, "content", response) if not isinstance(response, str) else response
    return not content or not content.strip()

def _handle_empty_response(self, original_len: int) -> bool:
    """返回True=已处理(continue循环), False=无法截断(报错退出)"""
    if original_len <= 4:
        logger.error("[run_stream] 历史太短无法截断，退出重试循环")
        return False
    kept_head = self.conversation_history[:2]
    kept_tail = self.conversation_history[-2:]
    seen_ids: Set[int] = set()
    deduped = []
    for item in kept_head + kept_tail:
        if id(item) not in seen_ids:
            seen_ids.add(id(item))
            deduped.append(item)
    self.conversation_history = deduped
    self.message_builder.conversation_history = deduped
    logger.warning(f"[run_stream] 空响应截断历史: {original_len}→{len(deduped)}")
    return True
```

    self, tool_name: str, tool_params: Dict[str, Any],
    step_count: int, chunk_buffer: ChunkBuffer
) -> AsyncGenerator[Union[Step, SSEEvent], None]:
    chunk_buffer.clear()
    # ThoughtStep
    thought_step = StepFactory.create_thought(
        f"执行工具: {tool_name}", tool_name=tool_name, params=tool_params
    )
    yield thought_step

    # 中断检查
    if await self._should_interrupt_and_yield(step_count, self.task_id, "action_before_exec"):
        return

    # 执行工具
    observation = await self._execute_tool_step(tool_name, tool_params)

    # Observation
    obs_step = StepFactory.create_observation(observation)
    yield obs_step

    # pending calls（异步回调）
    if hasattr(self, "_pending_calls") and self._pending_calls:
        pending_result = await self._execute_pending_calls(step_count)
        async for chunk in pending_result.chunks:
            yield chunk

```python
class TaskLifecycleManager:
    """管理running_tasks字典的注册、查询、清理。消除generate_sse_stream/cancel/pause/resume中的重复"""
    def __init__(self, running_tasks: Dict[str, Any], lock: asyncio.Lock):
        self._tasks = running_tasks
        self._lock = lock

    async def register(self, task_id: str, ai_service: Any) -> None:
        async with self._lock:
            self._tasks[task_id] = {
                "status": "running",
                "cancelled": False,
                "paused": False,
                "created_at": datetime.now(),
                "ai_service": ai_service,
                "_task": asyncio.current_task(),
            }

    async def cancel(self, task_id: str) -> bool:
        async with self._lock:
            task = self._tasks.get(task_id)
            if task is None:
                return False
            task["cancelled"] = True
            task["status"] = "cancelled"
            py_task = task.get("_task")
            if py_task and not py_task.done():
                py_task.cancel()
            return True

    async def cleanup(self, task_id: str) -> bool:
        """返回True=已清理, False=保留(如cancelled记录)"""
        async with self._lock:
            task = self._tasks.get(task_id)
            if task is None:
                return False
            if task.get("status") == "cancelled":
                return False  # 保留cancelled记录供前端查询
            del self._tasks[task_id]
            return True

    def is_cancelled(self, task_id: str) -> bool:
        task = self._tasks.get(task_id)
        return bool(task and task.get("cancelled"))
```

```python
async def generate_sse_stream(
    messages: List[Dict[str, str]],
    intent_type: str,
    confidence: float,
    session_id: Optional[str] = None,
    next_step: Optional[Dict[str, Any]] = None,
    current_execution_steps: Optional[List[Dict[str, Any]]] = None,
    ai_service: Optional[Any] = None,
    task_id: Optional[str] = None,
    running_tasks: Optional[Dict[str, Any]] = None,
    running_tasks_lock: Optional[asyncio.Lock] = None,
) -> AsyncGenerator[str, None]:
    manager = TaskLifecycleManager(running_tasks or {}, running_tasks_lock or asyncio.Lock())
    llm_call_count = 0
    agent_llm_holder: Dict[str, int] = {}
    task_id = task_id or str(uuid.uuid4())
    await manager.register(task_id, ai_service)

    try:
        # Prompt日志
        if intent_type not in ("", "generic", "chat"):
            await _log_prompts(messages, intent_type, confidence, session_id, task_id)

        # 安全检查
        security_error = await _perform_security_check(
            messages, next_step, session_id, current_execution_steps, ai_service
        )
        if security_error:
            yield security_error
            return

        # 中断/暂停检查
        if _is_interrupted(task_id, running_tasks):
            yield _build_interrupted_sse(task_id)
            return
        if _is_paused(task_id, running_tasks):
            yield _build_paused_sse(task_id)
            return

        # Agent分发
        try:
            agent = AgentFactory.create(intent_type, ai_service)
        except ValueError:
            agent = _GenericAgent(ai_service)

        async for chunk in _run_sse_stream(agent, messages, session_id, task_id, intent_type):
            yield chunk

    except asyncio.CancelledError:
        await _handle_client_disconnect(task_id, session_id, current_execution_steps)
    except Exception as e:
        await _handle_stream_error(e, task_id, session_id, current_execution_steps)
    finally:
        await _cleanup_task(task_id, manager, agent_llm_holder, llm_call_count)
```

```python
async def _run_sse_stream(
    agent: BaseAgent, messages: List[Dict[str, str]],
    session_id: Optional[str], task_id: str, intent_type: str,
) -> AsyncGenerator[str, None]:
    step_count = 0
    try:
        async for step in agent.run_stream(messages):
            sse_event = format_sse_event(step.type, step.to_dict())
            yield sse_event
            if step.type in ("final", "error", "interrupted"):
                break
            step_count += 1
    except asyncio.CancelledError:
        logger.info(f"[_run_sse_stream] 任务{task_id}被中断")
        interrupted_step = create_incident_data(task_id, "interrupted")
        yield format_sse_event("interrupted", interrupted_step)
    except Exception as e:
        logger.error(f"[_run_sse_stream] 流异常: {e}")
        error_step = StepFactory.create_error_step(str(e))
        yield format_sse_event("error", error_step.to_dict())
    finally:
        if session_id:
            try:
                await save_execution_steps_to_db(session_id, agent.steps)
            except Exception as e:
                logger.warning(f"保存execution_steps失败: {e}")
```

**组件2: `AssistantMessageIdAllocator`类**

```python
class AssistantMessageIdAllocator:
    """为assistant消息分配唯一ID。save_execution_steps和save_message复用"""
    def __init__(self, user_ids: Dict[str, int], lock: threading.Lock):
        self._user_ids = user_ids
        self._assistant_ids: Dict[str, int] = {}
        self._lock = lock

    def allocate(self, session_id: str, conn: sqlite3.Connection) -> Tuple[int, bool]:
        """返回 (message_id, is_new)"""
        with self._lock:
            user_id = self._user_ids.get(session_id)

        if user_id is not None:
            expected = user_id + 1
        else:
            cursor = conn.cursor()
            cursor.execute(
                "SELECT id FROM chat_messages WHERE session_id=? AND role='user' ORDER BY id DESC LIMIT 1",
                (session_id,),
            )
            row = cursor.fetchone()
            expected = (row["id"] + 1) if row else 1

        cursor = conn.cursor()
        cursor.execute("SELECT id, role FROM chat_messages WHERE id=?", (expected,))
        existing = cursor.fetchone()
        if existing and existing["role"] == "assistant":
            return expected, False  # 已存在，更新即可
        if existing and existing["role"] != "assistant":
            cursor.execute(
                "SELECT id FROM chat_messages WHERE session_id=? ORDER BY id DESC LIMIT 1",
                (session_id,),
            )
            max_row = cursor.fetchone()
            expected = (max_row["id"] + 1) if max_row else 1

        with self._lock:
            self._assistant_ids[session_id] = expected
        return expected, True  # 新消息
```

```python
import functools
from typing import Callable, Any

def with_timeout(deadline_attr: str = "deadline") -> Callable:
    """装饰器：函数内通过self或局部变量获取截止时间，超时提前返回"""
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        def wrapper(*args: Any, **kwargs: Any) -> Any:
            deadline = kwargs.get(deadline_attr) or getattr(args[0], deadline_attr, None) if args else None
            if deadline is None:
                return func(*args, **kwargs)
            if time.monotonic() > deadline:
                logger.warning(f"[{func.__name__}] 超时，跳过执行")
                return []
            return func(*args, **kwargs)
        return wrapper
    return decorator
```

**组件3: `_resolve_name_conflict`冲突解决**

```python
def _resolve_name_conflict(
    new_path: Path, strategy: str, max_attempts: int = 100,
) -> Tuple[Path, bool]:
    """返回 (最终路径, 是否解决)。strategy: skip/overwrite/rename"""
    if not new_path.exists():
        return new_path, True

    if strategy == "skip":
        return new_path, False  # resolved=False → 跳过

    if strategy == "overwrite":
        try:
            new_path.unlink()
            return new_path, True
        except Exception:
            return new_path, False

    # strategy == "rename": 加数字后缀
    counter = 1
    while new_path.exists() and counter <= max_attempts:
        stem = new_path.stem
        suffix = new_path.suffix
        new_path = new_path.parent / f"{stem}_{counter}{suffix}"
        counter += 1

    if counter > max_attempts:
        logger.error(f"冲突解决失败，超过{max_attempts}次尝试")
        return new_path, False
    return new_path, True
```

```python
            execute_with_safety,
            operation_id=operation_id,
            operation_func=lambda: _rename_file(file_path, resolved_path),
        )
        actually_renamed = result.get("success", False)
        return {
            "file": str(file_path), "original_name": original_name,
            "new_name": resolved_path.name, "status": "renamed" if actually_renamed else "failed",
            "actually_renamed": actually_renamed,
            "operation_id": operation_id,
        }
    except Exception as e:
        return {"file": str(file_path), "status": "error", "reason": str(e),
                "actually_renamed": False}

def _rename_file(src: Path, dst: Path) -> bool:
    if dst.exists():
        dst.unlink()
    dst.parent.mkdir(parents=True, exist_ok=True)
    shutil.move(str(src), str(dst))
    return True
```

```python
async def batch_rename_impl(
    directory: str,
    pattern: str,
    replacement: str,
    recursive: bool = False,
    conflict_strategy: str = "rename",
    preview: bool = False,
    task_id: Optional[str] = None,
    timeout: int = 30,
) -> Dict[str, Any]:
    # 参数校验
    is_valid, error_msg = validate_path(directory)
    if not is_valid:
        return {"status": "error", "message": error_msg}

    dir_path = Path(directory)
    if not dir_path.is_dir():

    deadline = time.monotonic() + timeout

    # 收集文件
    files, timed_out = await _collect_files(dir_path, recursive, deadline)

    # 逐文件处理

        operations.append(op)

    error_count = sum(1 for op in operations if op.get("status") == "error")

    return {
        "status": "ok",
        "directory": directory,
        "pattern": pattern,
        "replacement": replacement,
        "total": len(files),
        "renamed": renamed_count,
        "skipped": skipped_count,
        "errors": error_count,
        "timed_out": timed_out,
        "operations": operations if preview else None,  # 仅preview时返回详细列表
```
