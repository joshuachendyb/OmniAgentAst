```python
    """清理过期任务"""
    now = datetime.now()
    async with running_tasks_lock:
        expired_tasks = [
            task_id for task_id, task in running_tasks.items()
            if task.get("created_at") and now - task["created_at"] > TASK_TIMEOUT
        ]
        # 同时清理已取消超过1小时的任务
        cancelled_tasks = [
            task_id for task_id, task in running_tasks.items()
            if task.get("cancelled") and task.get("interrupt_time")
            and now - datetime.fromisoformat(task["interrupt_time"]) > timedelta(hours=1)
        ]
        tasks_to_remove = expired_tasks + cancelled_tasks
        for task_id in tasks_to_remove:
            del running_tasks[task_id]
        if tasks_to_remove:
            logger.info(f"清理了 {len(tasks_to_remove)} 个过期/已取消任务")
```

```typescript
// 等待中断事件的智能策略
const waitForInterruptEvent = async (maxWaitTime = 3000, checkInterval = 200): Promise<boolean> => {
    const startTime = Date.now();
    let hasReceivedEvent = false;
    
    while (Date.now() - startTime < maxWaitTime) {
        if (hasReceivedInterruptEventRef.current) {
            console.log("[中断] 已收到 interrupted 事件");
            hasReceivedEvent = true;
            break;
        }
        await new Promise(resolve => setTimeout(resolve, checkInterval));
    }
    
    if (!hasReceivedEvent) {
        console.warn(`[中断] 在 ${maxWaitTime}ms 内未收到 interrupted 事件，继续执行`);
    }
    
    return hasReceivedEvent;
};
```

**修改中断处理**:

### 8. 添加中断监控日志

**文件**: 新建 `frontend/src/utils/interruptMonitor.ts`
**内容**:
```typescript
class InterruptMonitor {
    private static instance: InterruptMonitor;
    private logs: Array<{timestamp: number; event: string; data: any}> = [];
    
    static getInstance() {
        if (!InterruptMonitor.instance) {
            InterruptMonitor.instance = new InterruptMonitor();
        }
        return InterruptMonitor.instance;
    }
    
    log(event: string, data?: any) {
        const logEntry = {
            timestamp: Date.now(),
            event,
            data
        };
        this.logs.push(logEntry);
        console.log(`[中断监控] ${event}`, data);
        
        // 保持最近100条日志
        if (this.logs.length > 100) {
            this.logs = this.logs.slice(-100);
        }
    }
    
    getLogs() {
        return [...this.logs];
    }
    
    clear() {
        this.logs = [];
    }
    
    // 记录中断过程
    startInterrupt(taskId: string) {
        this.log('中断开始', { taskId, time: Date.now() });
    }
    
    apiCalled(taskId: string) {
        this.log('中断API调用', { taskId, time: Date.now() });
    }
    
    eventReceived(taskId: string) {
        this.log('收到中断事件', { taskId, time: Date.now() });
    }
    
    disconnectCalled(taskId: string) {
        this.log('断开连接', { taskId, time: Date.now() });
    }
    
    interruptComplete(taskId: string, success: boolean, error?: string) {
        this.log('中断完成', { taskId, success, error, time: Date.now() });
    }
}

export const interruptMonitor = InterruptMonitor.getInstance();
```

```python
import pytest
import asyncio
from unittest.mock import AsyncMock, MagicMock
from app.services.react_sse_wrapper import generate_sse_stream, cancel_task

@pytest.mark.asyncio
async def test_task_interruption():
    """测试任务中断功能"""
    task_id = "test_task_123"
    
    # 模拟一个长时间运行的任务
    async def simulate_long_running_task(task_id):
        # 注册任务
        running_tasks = {}
        running_tasks[task_id] = {
            "status": "running",
            "cancelled": False,
            "created_at": datetime.now()
        }
        
        # 模拟任务运行
        for i in range(10):
            await asyncio.sleep(0.1)
            # 检查中断
            if running_tasks.get(task_id, {}).get("cancelled", False):
                return "interrupted"
        
        return "completed"
    
    # 启动任务
    task = asyncio.create_task(simulate_long_running_task(task_id))
    
    # 等待一段时间后中断
    await asyncio.sleep(0.5)
    result = await cancel_task(task_id)
    
    assert result["success"] == True
    assert result["task_status"] == "cancelled"
    
    # 等待任务结束
    task_result = await task
    assert task_result == "interrupted"
```
